94 /*FUNCTION**********************************************************************
95  *
96  * Function Name : SMC_HAL_SetMode
97  * Description : Config the power mode
98  * This function configures the power mode base on configuration structure, if
99  * could not switch to the target mode directly, this function could check
100  * internally and choose the right path.
101  *
102  *END*/
103 smc_hal_error_code_t SMC_HAL_SetMode(SMC_Type * base,
104  const smc_power_mode_config_t *powerModeConfig)
105 {
106  volatile uint32_t dummyRead;
107  smc_stop_mode_t stopMode;
108  power_mode_stat_t curPowerMode;
109 
110  curPowerMode = SMC_HAL_GetStat(base);
111 
112 #if FSL_FEATURE_SMC_HAS_LPWUI
113 
114  if (((kPowerModeVlpr | kPowerModeVlpw | kPowerModeVlps) & (uint32_t)powerModeConfig->powerModeName)
115  && (powerModeConfig->lpwuiOptionValue != (smc_lpwui_option_t)SMC_BRD_PMCTRL_LPWUI(base)))
116  {
117  SMC_HAL_SetPowerModeRun(base);
118  SMC_BWR_PMCTRL_LPWUI(base, powerModeConfig->lpwuiOptionValue);
119  /*
120  * Why set to VLPR here:
121  * 1. If target mode is VLPS, then LPWUI will be bypassed if enter VLPS
122  * from RUN directly. So first change to VLPR to make sure LPWUI is
123  * not bypassed.
124  * 2. If target mode is VLPR/VLPW, it is OK to set to VLPR first.
125  */
126  SMC_HAL_SetPowerModeVlpr(base);
127  curPowerMode = kStatVlpr;
128  }
129 #endif
130 
131  /*
132  * Step 1: If could not reach target mode directly, change to the mode that
133  * could reach.
134  */
135  switch (curPowerMode)
136  {
137  // If current mode is RUN, target mode is VLPW, change to VLPR first.
138  case kStatRun:
139  if (kPowerModeVlpw & (uint32_t)powerModeConfig->powerModeName)
140  {
141  SMC_HAL_SetPowerModeVlpr(base);
142  }
143  break;
144  // If current mode is VLPR, target mode is HSRUN/STOP/WAIT, change to RUN first.
145  case kStatVlpr:
146  if ((kPowerModeStop
147  | kPowerModeWait
148 #if FSL_FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE
149  | kPowerModeHsrun
150 #endif
151  ) & (uint32_t)powerModeConfig->powerModeName)
152  {
153  SMC_HAL_SetPowerModeRun(base);
154  }
155  break;
156 #if FSL_FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE
157 
161  case kStatHsrun:
162  if (!((kPowerModeHsrun | kPowerModeRun) & (uint32_t)powerModeConfig->powerModeName))
163  {
164  SMC_HAL_SetPowerModeRun(base);
165 
166  if (kPowerModeVlpw & (uint32_t)powerModeConfig->powerModeName)
167  {
168  SMC_HAL_SetPowerModeVlpr(base);
169  }
170  }
171  break;
172 #endif
173  default:
174  break;
175  }
176 
177  // Step 2: Change to target mode.
178  switch (powerModeConfig->powerModeName)
179  {
180  case kPowerModeRun:
181  SMC_HAL_SetPowerModeRun(base);
182  break;
183 
184  case kPowerModeVlpr:
185  SMC_HAL_SetPowerModeVlpr(base);
186  break;
187 
188 #if FSL_FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE
189  case kPowerModeHsrun:
190  SMC_HAL_SetPowerModeHsrun(base);
191  break;
192 #endif
193 
195  case kPowerModeWait:
196  case kPowerModeVlpw:
198  SCB->SCR &= ~SCB_SCR_SLEEPDEEP_Msk;
199  /*
200  * Ensure that all register writes associated with setting up the
201  * low power mode being entered have completed before the MCU enters
202  * the low power mode.
203  */
204  dummyRead = SMC_RD_PMCTRL(base);
205  dummyRead = dummyRead;
206  __WFI();
207  break;
208 
210  case kPowerModeStop:
211  case kPowerModeVlps:
212 #if FSL_FEATURE_SMC_HAS_LOW_LEAKAGE_STOP_MODE
213  case kPowerModeLls:
214 #endif
215  case kPowerModeVlls:
216 #if FSL_FEATURE_SMC_HAS_PSTOPO
217  SMC_BWR_STOPCTRL_PSTOPO(base, powerModeConfig->pstopOptionValue);
218 #endif
219 #if FSL_FEATURE_SMC_HAS_LPOPO
220  SMC_BWR_STOPCTRL_LPOPO(base, powerModeConfig->lpoOptionValue);
221 #endif
222 #if FSL_FEATURE_SMC_HAS_PORPO
223  SMC_SET_PORPO(base, powerModeConfig->porOptionValue);
224 #endif
225 #if FSL_FEATURE_SMC_HAS_RAM2_POWER_OPTION
226  SMC_SET_RAM2PO(base, powerModeConfig->ram2OptionValue);
227 #endif
228  if (kPowerModeStop == powerModeConfig->powerModeName)
229  {
230  stopMode = kSmcStop;
231  }
232  else if (kPowerModeVlps == powerModeConfig->powerModeName)
233  {
234  stopMode = kSmcVlps;
235  }
236 #if FSL_FEATURE_SMC_HAS_LOW_LEAKAGE_STOP_MODE
237  else if (kPowerModeLls == powerModeConfig->powerModeName)
238  {
239  stopMode = kSmcLls;
240  SMC_SET_STOP_SUBMODE(base, powerModeConfig->stopSubMode);
241  }
242 #endif
243  else
244  {
245  stopMode = kSmcVlls;
246  SMC_SET_STOP_SUBMODE(base, powerModeConfig->stopSubMode);
247  }
248 
249  SMC_BWR_PMCTRL_STOPM(base, stopMode);
251  SCB->SCR |= SCB_SCR_SLEEPDEEP_Msk;
252  /*
253  * Ensure that all register writes associated with setting up the
254  * low power mode being entered have completed before the MCU enters
255  * the low power mode.
256  */
257  dummyRead = SMC_RD_PMCTRL(base);
258  __WFI();
259  break;
260 
261  default:
262  break;
263  }
264 
265  return kSmcHalSuccess;
266 }

268 /*FUNCTION**********************************************************************
269  *
270  * Function Name : SMC_HAL_GetStat
271  * Description : Get the current power mode stat
272  * This function will return the current power mode stat. Once application is
273  * switching the power mode, it should always check the stat to make sure it
274  * runs into the specified mode or not. Also application will need to check
275  * this mode before switching to certain mode. The system will require that
276  * only certain mode could switch to other specific mode. Refer to the
277  * reference manual for details. Refer to _power_mode_stat for the meaning
278  * of the power stat
279  *
280  *END*/
281 power_mode_stat_t SMC_HAL_GetStat(SMC_Type * base)
282 {
283  return (power_mode_stat_t)SMC_RD_PMSTAT(base);
284 }
