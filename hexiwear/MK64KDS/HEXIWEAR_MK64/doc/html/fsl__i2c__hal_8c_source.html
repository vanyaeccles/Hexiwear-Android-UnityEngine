145 /*FUNCTION**********************************************************************
146  *
147  * Function Name : I2C_HAL_SetBaudRate
148  * Description : Sets the I2C bus frequency for master transactions.
149  *
150  *END*/
151 void I2C_HAL_SetBaudRate(I2C_Type * base,
152  uint32_t sourceClockInHz,
153  uint32_t kbps,
154  uint32_t * absoluteError_Hz)
155 {
156  uint32_t mult, i, multiplier, computedRate, absError;
157  uint32_t hz = kbps * 1000u;
158  uint32_t bestError = 0xffffffffu;
159  uint32_t bestMult = 0u;
160  uint32_t bestIcr = 0u;
161 
165  for (mult = 0u; (mult <= 2u) && (bestError != 0); ++mult)
166  {
167  multiplier = 1u << mult;
168 
170  for (i = 0u; i < ARRAY_SIZE(kI2CDividerTable); ++i)
171  {
172  computedRate = sourceClockInHz / (multiplier * kI2CDividerTable[i].sclDivider);
173  absError = hz > computedRate ? hz - computedRate : computedRate - hz;
174 
175  if (absError < bestError)
176  {
177  bestMult = mult;
178  bestIcr = kI2CDividerTable[i].icr;
179  bestError = absError;
180 
183  if (absError == 0)
184  {
185  break;
186  }
187  }
188  }
189  }
190 
192  if (absoluteError_Hz)
193  {
194  *absoluteError_Hz = bestError;
195  }
196 
198  I2C_WR_F(base, I2C_F_MULT(bestMult) | I2C_F_ICR(bestIcr));
199 }

201 /*FUNCTION**********************************************************************
202  *
203  * Function Name : I2C_HAL_SendStart
204  * Description : Send a START or Repeated START signal on the I2C bus.
205  * This function is used to initiate a new master mode transfer by sending the
206  * START signal. It is also used to send a Repeated START signal when a transfer
207  * is already in progress.
208  *
209  *END*/
210 void I2C_HAL_SendStart(I2C_Type * base)
211 {
213  if (I2C_BRD_C1_MST(base))
214  {
215 #if FSL_FEATURE_I2C_HAS_ERRATA_6070
216 
221  uint32_t savedMult = 0;
222  if (I2C_BRD_F_MULT(base) != 0)
223  {
224  savedMult = I2C_BRD_F_MULT(base);
225  I2C_BWR_F_MULT(base, 0U);
226  }
227 #endif
230  I2C_BWR_C1_RSTA(base, 1U);
231 
232 #if FSL_FEATURE_I2C_HAS_ERRATA_6070
233  if (savedMult)
234  {
235  I2C_BWR_F_MULT(base, savedMult);
236  }
237 #endif
238  }
239  else
240  {
242  I2C_SET_C1(base, I2C_C1_MST_MASK | I2C_C1_TX_MASK);
243  }
244 }

246 /*FUNCTION**********************************************************************
247  *
248  * Function Name : I2C_HAL_SendStop
249  * Description : Sends a STOP signal on the I2C bus.
250  * This function changes the direction to receive.
251  *
252  *END*/
253 i2c_status_t I2C_HAL_SendStop(I2C_Type * base)
254 {
255  assert(I2C_BRD_C1_MST(base) == 1);
256  uint32_t i = 0;
257 
259  I2C_CLR_C1(base, I2C_C1_MST_MASK | I2C_C1_TX_MASK);
260 
262  while(I2C_HAL_GetStatusFlag(base, kI2CBusBusy))
263  {
264  if (++i == 0x400U)
265  {
267  return kStatus_I2C_StopSignalFail;
268  }
269  else
270  {
271  __asm("nop");
272  }
273  }
274 
275  return kStatus_I2C_Success;
276 }

314  *
315  * Function Name : I2C_HAL_ReadByteBlocking
316  * Description : Returns the last byte of data read from the bus and initiate
317  * another read. It will wait till the transfer is actually completed.
318  *
319  *END*/
320 uint8_t I2C_HAL_ReadByteBlocking(I2C_Type * base)
321 {
323  uint8_t byte = I2C_RD_D(base);
324 
326  while (!I2C_BRD_S_IICIF(base))
327  {}
328 
330  I2C_BWR_S_IICIF(base, 0x1U);
331 
332  return byte;
333 }

335 /*FUNCTION**********************************************************************
336  *
337  * Function Name : I2C_HAL_WriteByteBlocking
338  * Description : Writes one byte of data to the I2C bus and wait till that
339  * byte is transfered successfully.
340  *
341  *END*/
342 bool I2C_HAL_WriteByteBlocking(I2C_Type * base, uint8_t byte)
343 {
344 #if FSL_FEATURE_I2C_HAS_DOUBLE_BUFFERING
345  while (!I2C_BRD_S2_EMPTY(base))
346  {}
347 #endif
348 
350  I2C_WR_D(base, byte);
351 
353  while (!I2C_BRD_S_IICIF(base))
354  {}
355 
357  I2C_BWR_S_IICIF(base, 0x1U);
358 
360  return !I2C_BRD_S_RXAK(base);
361 }

363 /*FUNCTION**********************************************************************
364  *
365  * Function Name : I2C_HAL_MasterReceiveDataPolling
366  * Description : Performs a polling receive transaction on the I2C bus.
367  *
368  *END*/
369 i2c_status_t I2C_HAL_MasterReceiveDataPolling(I2C_Type * base,
370  uint16_t slaveAddr,
371  const uint8_t * cmdBuff,
372  uint32_t cmdSize,
373  uint8_t * rxBuff,
374  uint32_t rxSize)
375 {
376  int32_t i;
377  uint8_t directionBit, address;
378 
380  if (I2C_BRD_C1_MST(base))
381  {
382  return kStatus_I2C_Busy;
383  }
384 
387  directionBit = (cmdBuff) ? 0x0U : 0x1U;
388 
389  address = (uint8_t)slaveAddr << 1U;
390 
392  I2C_HAL_SetDirMode(base, kI2CSend);
393 
395  I2C_HAL_SendStart(base);
396 
398  if (!I2C_HAL_WriteByteBlocking(base, address | directionBit))
399  {
401  I2C_HAL_SendStop(base);
402  return kStatus_I2C_ReceivedNak;
403  }
404 
406  if (cmdBuff != NULL)
407  {
408  while (cmdSize--)
409  {
410  if (!I2C_HAL_WriteByteBlocking(base, *cmdBuff--))
411  {
413  I2C_HAL_SendStop(base);
414  return kStatus_I2C_ReceivedNak;
415  }
416  }
417 
419  I2C_HAL_SendStart(base);
420 
422  if (!I2C_HAL_WriteByteBlocking(base, address | 1U))
423  {
425  I2C_HAL_SendStop(base);
426  return kStatus_I2C_ReceivedNak;
427  }
428  }
429 
431  I2C_HAL_SetDirMode(base, kI2CReceive);
432 
434  if (rxSize == 0x1U)
435  {
436  I2C_HAL_SendNak(base);
437  }
438  else
439  {
440  I2C_HAL_SendAck(base);
441  }
442 
444  I2C_HAL_ReadByteBlocking(base);
445 
447  for(i=rxSize-1; i>=0; i--)
448  {
449  switch (i)
450  {
451  case 0x0U:
453  I2C_HAL_SendStop(base);
454  break;
455  case 0x1U:
457  I2C_HAL_SendNak(base);
458  break;
459  default :
460  I2C_HAL_SendAck(base);
461  break;
462  }
463 
465  if (i==0)
466  {
467  *rxBuff++ = I2C_HAL_ReadByte(base);
468  }
469  else
470  {
471  *rxBuff++ = I2C_HAL_ReadByteBlocking(base);
472  }
473  }
474 
475  return kStatus_I2C_Success;
476 }

478 /*FUNCTION**********************************************************************
479  *
480  * Function Name : I2C_HAL_MasterSendDataPolling
481  * Description : Performs a polling send transaction on the I2C bus.
482  *
483  *END*/
484 i2c_status_t I2C_HAL_MasterSendDataPolling(I2C_Type * base,
485  uint16_t slaveAddr,
486  const uint8_t * cmdBuff,
487  uint32_t cmdSize,
488  const uint8_t * txBuff,
489  uint32_t txSize)
490 {
491  uint16_t slaveAddress;
492  slaveAddress = (slaveAddr << 1U) & 0x00FFU;
493 
495  if (I2C_BRD_C1_MST(base))
496  {
497  return kStatus_I2C_Busy;
498  }
499 
501  I2C_HAL_SetDirMode(base, kI2CSend);
502 
504  I2C_HAL_SendStart(base);
505 
507  if (!I2C_HAL_WriteByteBlocking(base, slaveAddress))
508  {
510  I2C_HAL_SendStop(base);
511  return kStatus_I2C_ReceivedNak;
512  }
513 
515  if (cmdBuff != NULL)
516  {
517  while (cmdSize--)
518  {
519  if (!I2C_HAL_WriteByteBlocking(base, *cmdBuff--))
520  {
522  I2C_HAL_SendStop(base);
523  return kStatus_I2C_ReceivedNak;
524  }
525  }
526  }
527 
529  while (txSize--)
530  {
531  if (!I2C_HAL_WriteByteBlocking(base, *txBuff++))
532  {
534  I2C_HAL_SendStop(base);
535  return kStatus_I2C_ReceivedNak;
536  }
537  }
538 
540  I2C_HAL_SendStop(base);
541 
543  I2C_HAL_SetDirMode(base, kI2CReceive);
544 
545  return kStatus_I2C_Success;
546 }

548 /*FUNCTION**********************************************************************
549  *
550  * Function Name : I2C_HAL_SlaveSendDataPolling
551  * Description : Send out multiple bytes of data using polling method.
552  *
553  *END*/
554 i2c_status_t I2C_HAL_SlaveSendDataPolling(I2C_Type * base, const uint8_t* txBuff, uint32_t txSize)
555 {
556 #if FSL_FEATURE_I2C_HAS_START_STOP_DETECT
557 
558  while(!I2C_HAL_GetStartFlag(base))
559  {}
560  I2C_HAL_ClearStartFlag(base);
561 #endif
562 
563  while(!I2C_HAL_GetStatusFlag(base, kI2CAddressAsSlave))
564  {}
565 
567  while(!I2C_HAL_IsIntPending(base))
568  {}
570  I2C_HAL_ClearInt(base);
571 
573  if (I2C_HAL_GetStatusFlag(base, kI2CSlaveTransmit))
574  {
576  I2C_HAL_SetDirMode(base, kI2CSend);
577  }
578  else
579  {
581  I2C_HAL_SetDirMode(base, kI2CReceive);
582 
584  I2C_HAL_ReadByte(base);
585  }
586 
588  while(txSize--)
589  {
591  I2C_HAL_WriteByte(base, *txBuff++);
592 
594  while(!I2C_HAL_IsIntPending(base))
595  {}
596 
598  I2C_HAL_ClearInt(base);
599 
601  if ((I2C_HAL_GetStatusFlag(base, kI2CReceivedNak)) && (txSize != 0))
602  {
603  return kStatus_I2C_ReceivedNak;
604  }
605  }
607  I2C_HAL_SetDirMode(base, kI2CReceive);
608 
610  I2C_HAL_ReadByte(base);
611 
612  return kStatus_I2C_Success;
613 }
