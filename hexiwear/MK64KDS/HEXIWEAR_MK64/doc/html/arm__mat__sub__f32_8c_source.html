72 arm_status arm_mat_sub_f32(
73  const arm_matrix_instance_f32 * pSrcA,
74  const arm_matrix_instance_f32 * pSrcB,
75  arm_matrix_instance_f32 * pDst)
76 {
77  float32_t *pIn1 = pSrcA->pData;
78  float32_t *pIn2 = pSrcB->pData;
79  float32_t *pOut = pDst->pData;
81 #ifndef ARM_MATH_CM0_FAMILY
82 
83  float32_t inA1, inA2, inB1, inB2, out1, out2;
85 #endif // #ifndef ARM_MATH_CM0_FAMILY
86 
87  uint32_t numSamples;
88  uint32_t blkCnt;
89  arm_status status;
91 #ifdef ARM_MATH_MATRIX_CHECK
92 
93  if((pSrcA->numRows != pSrcB->numRows) ||
94  (pSrcA->numCols != pSrcB->numCols) ||
95  (pSrcA->numRows != pDst->numRows) || (pSrcA->numCols != pDst->numCols))
96  {
98  status = ARM_MATH_SIZE_MISMATCH;
99  }
100  else
101 #endif /* #ifdef ARM_MATH_MATRIX_CHECK */
102  {
104  numSamples = (uint32_t) pSrcA->numRows * pSrcA->numCols;
105 
106 #ifndef ARM_MATH_CM0_FAMILY
107 
111  blkCnt = numSamples >> 2u;
112 
115  while(blkCnt > 0u)
116  {
120  inA1 = pIn1[0];
121 
123  inB1 = pIn2[0];
124 
126  inA2 = pIn1[1];
127 
129  out1 = inA1 - inB1;
130 
132  inB2 = pIn2[1];
133 
135  inA1 = pIn1[2];
136 
138  out2 = inA2 - inB2;
139 
141  inB1 = pIn2[2];
142 
144  pOut[0] = out1;
145  pOut[1] = out2;
146 
148  inA2 = pIn1[3];
149 
151  inB2 = pIn2[3];
152 
154  out1 = inA1 - inB1;
155 
156 
158  out2 = inA2 - inB2;
159 
161  pOut[2] = out1;
162 
164  pOut[3] = out2;
165 
166 
168  pIn1 += 4u;
169  pIn2 += 4u;
170  pOut += 4u;
171 
173  blkCnt--;
174  }
175 
178  blkCnt = numSamples % 0x4u;
179 
180 #else
181 
185  blkCnt = numSamples;
186 
187 #endif /* #ifndef ARM_MATH_CM0_FAMILY */
188 
189  while(blkCnt > 0u)
190  {
193  *pOut++ = (*pIn1++) - (*pIn2++);
194 
196  blkCnt--;
197  }
198 
200  status = ARM_MATH_SUCCESS;
201  }
202 
204  return (status);
205 }
