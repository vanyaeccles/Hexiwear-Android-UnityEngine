321 #if configUSE_TICKLESS_IDLE == 1
322 __attribute__((weak)) void vPortSuppressTicksAndSleep(TickType_t xExpectedIdleTime) {
323  unsigned long ulReloadValue, ulCompleteTickPeriods, ulCompletedSysTickIncrements;
324  TickCounter_t tmp;
325  bool tickISRfired;
326 
328  if(xExpectedIdleTime>xMaximumPossibleSuppressedTicks) {
329  xExpectedIdleTime = xMaximumPossibleSuppressedTicks;
330  }
331 
337  DISABLE_TICK_COUNTER();
338 
343  GET_TICK_CURRENT_VAL(&tmp);
344  ulReloadValue = tmp+(UL_TIMER_COUNTS_FOR_ONE_TICK*(xExpectedIdleTime-1UL));
345  if (ulReloadValue>ulStoppedTimerCompensation) {
346  ulReloadValue -= ulStoppedTimerCompensation;
347  }
348 
352  TICKLESS_DISABLE_INTERRUPTS();
353 
357  if (eTaskConfirmSleepModeStatus()==eAbortSleep) {
358  ENABLE_TICK_COUNTER();
359  TICKLESS_ENABLE_INTERRUPTS();
360  } else {
361 #if configUSE_LP_TIMER
362  DisableDevice();
363  ClearInterruptFlag();
364  WriteCompareReg(xExpectedIdleTime-1);
365  EnableDevice();
366 #else
367  SET_TICK_DURATION(ulReloadValue);
368  RESET_TICK_COUNTER_VAL();
369  ENABLE_TICK_COUNTER();
370  TICK_INTERRUPT_FLAG_RESET();
371 #endif
372 
382  __asm volatile("dsb");
383  __asm volatile("wfi");
384  __asm volatile("isb");
385  /* ----------------------------------------------------------------------------
386  * Here the CPU *HAS TO BE* low power mode, waiting to wake up by an interrupt
387  * ----------------------------------------------------------------------------*/
393  tickISRfired = TICK_INTERRUPT_HAS_FIRED();
394  DISABLE_TICK_COUNTER();
395  TICKLESS_ENABLE_INTERRUPTS();
396  if (tickISRfired) {
402  GET_TICK_CURRENT_VAL(&tmp);
403 #if COUNTS_UP
404  SET_TICK_DURATION((UL_TIMER_COUNTS_FOR_ONE_TICK-1UL)-tmp);
405 #else
406  SET_TICK_DURATION((UL_TIMER_COUNTS_FOR_ONE_TICK-1UL)-(ulReloadValue-tmp));
407 #endif
408 
414  ulCompleteTickPeriods = xExpectedIdleTime-1UL;
415  } else {
420  GET_TICK_CURRENT_VAL(&tmp);
421  ulCompletedSysTickIncrements = (xExpectedIdleTime*UL_TIMER_COUNTS_FOR_ONE_TICK)-tmp;
422 
424  ulCompleteTickPeriods = ulCompletedSysTickIncrements/UL_TIMER_COUNTS_FOR_ONE_TICK;
425 
427  SET_TICK_DURATION(((ulCompleteTickPeriods+1)*UL_TIMER_COUNTS_FOR_ONE_TICK)-ulCompletedSysTickIncrements);
428  }
429 
436  RESET_TICK_COUNTER_VAL();
437  portENTER_CRITICAL();
438  {
439  ENABLE_TICK_COUNTER();
440  vTaskStepTick(ulCompleteTickPeriods);
441  SET_TICK_DURATION(UL_TIMER_COUNTS_FOR_ONE_TICK-1UL);
442  }
443  portEXIT_CRITICAL();
444  }
445 }
446 #endif /* #if configUSE_TICKLESS_IDLE */

448 void vPortInitTickTimer(void) {
449 #if configUSE_TICKLESS_IDLE == 1
450 {
451 #if TICK_NOF_BITS==32
452  xMaximumPossibleSuppressedTicks = 0xffffffffUL/TIMER_COUNTS_FOR_ONE_TICK; /* 32bit timer register */
453 #elif TICK_NOF_BITS==24
454  xMaximumPossibleSuppressedTicks = 0xffffffUL/TIMER_COUNTS_FOR_ONE_TICK; /* 24bit timer register */
455 #elif TICK_NOF_BITS==16
456  xMaximumPossibleSuppressedTicks = 0xffffUL/TIMER_COUNTS_FOR_ONE_TICK; /* 16bit timer register */
457 #elif TICK_NOF_BITS==8
458  xMaximumPossibleSuppressedTicks = 0xffUL/TIMER_COUNTS_FOR_ONE_TICK; /* 8bit timer register */
459 #else
460  error "unknown configuration!"
461 #endif
462 #if configSYSTICK_USE_LOW_POWER_TIMER
463  ulStoppedTimerCompensation = configSTOPPED_TIMER_COMPENSATION/(configCPU_CLOCK_HZ/configSYSTICK_LOW_POWER_TIMER_CLOCK_HZ);
464 #else
465  ulStoppedTimerCompensation = configSTOPPED_TIMER_COMPENSATION/(configCPU_CLOCK_HZ/configSYSTICK_CLOCK_HZ);
466 #endif
467 }
468 #endif
469 #if configSYSTICK_USE_LOW_POWER_TIMER
470  SIM_SCGC5 |= SIM_SCGC5_LPTMR_MASK;
473  LPTMR0_CSR = (LPTMR_CSR_TCF_MASK | LPTMR_CSR_TPS(0x00));
475  LPTMR0_PSR = LPTMR_PSR_PRESCALE(0x00) |
476  LPTMR_PSR_PBYP_MASK |
477  LPTMR_PSR_PCS(0x01);
478  /*
479  * PBYP PCS
480  * ERCLK32 1 10
481  * LPO_1kHz 1 01
482  * ERCLK 0 00
483  * IRCLK 1 00
484  */
485  *(portNVIC_SYSPRI7) |= portNVIC_LP_TIMER_PRI;
487  NVIC_ISER |= NVIC_ISER_SETENA(0x10000000); /* 0xE000E100 <= 0x10000000 */
488 
490  LPTMR0_CSR = (LPTMR_CSR_TPS(0x00) | LPTMR_CSR_TEN_MASK);
491 #else
492  *(portNVIC_SYSPRI3) |= portNVIC_SYSTICK_PRI;
493 #endif
494 
495  SET_TICK_DURATION(TIMER_COUNTS_FOR_ONE_TICK-1UL);
496  RESET_TICK_COUNTER_VAL();
497  ENABLE_TICK_COUNTER();
498 }

559 void vPortEnterCritical(void) {
560 /*
561  * Disable interrupts before incrementing the count of critical section nesting.
562  * The nesting count is maintained so we know when interrupts should be
563  * re-enabled. Once interrupts are disabled the nesting count can be accessed
564  * directly. Each task maintains its own nesting count.
565  */
566  portDISABLE_INTERRUPTS();
567  portPOST_ENABLE_DISABLE_INTERRUPTS();
568  uxCriticalNesting++;
569  __asm volatile("dsb");
570  __asm volatile("isb");
571 }
572 /*-----------------------------------------------------------*/
573 void vPortExitCritical(void) {
578  uxCriticalNesting--;
579  if (uxCriticalNesting == 0) {
580  portENABLE_INTERRUPTS();
581  portPOST_ENABLE_DISABLE_INTERRUPTS();
582  }
583 }
