561 gui_status_t GuiDriver_LabelFormatCaption(guiLabel_t *label, uint8_t *caption, uint8_t *length, guiLabelCutFormat_t format)
562 {
563  uint8_t tmpLength;
564  uint8_t inLength;
565 
566  if(length == NULL)
567  {
568  inLength = strlen((char *)caption);
569  }
570  else
571  {
572  inLength = *length;
573  }
574 
575  if(inLength > (label->captionLength - 1))
576  {
577  tmpLength = label->captionLength - 1;
578  }
579  else
580  {
581  tmpLength = inLength;
582  }
583 
584  tmpLength = OLED_CharCount(label->dynamicArea.width, label->textProperties.font, caption, tmpLength);
585  if(tmpLength == 0)
586  {
587  return GUI_STATUS_ERROR;
588  }
589 
590  memcpy(label->caption, caption, tmpLength);
591  label->caption[tmpLength] = 0;
592 
593  if(tmpLength < inLength)
594  {
595  if(format == GUI_LABEL_CUT_FORMAT_DOTS)
596  {
597  if(tmpLength <= 2)
598  {
599  return GUI_STATUS_ERROR;
600  }
601 
602  label->caption[tmpLength-1] = '.';
603  label->caption[tmpLength-2] = '.';
604  tmpLength -= 2;
605  }
606  else if(format == GUI_LABEL_CUT_FORMAT_SPACE)
607  {
608  char *tmpPtr;
609  tmpPtr = strrchr((char *)label->caption, ' ');
610  if(
611  (tmpPtr != NULL) &&
612  (tmpPtr != (char *)label->caption)
613  )
614  {
615  *tmpPtr = 0;
616  tmpLength = tmpPtr - (char *)label->caption;
617  }
618  }
619  }
620 
621  if(length != NULL)
622  {
623  *length = tmpLength;
624  }
625 
626  return GUI_STATUS_SUCCESS;
627 }
