139 bool debounce_count (
140  bool dbcntm,
141  bool condition,
142  debounce_count_t* count,
143  debounce_count_t threshold
144  )
145 {
146  if ( 0 != condition )
147  {
148  // increment count up to, but not above the threshold
149  *count = (*count+1) < threshold ? *count+1 : threshold;
150  }
151 
152  else
153  {
154  if ( 0 != dbcntm )
155  {
156  *count = 0;
157  }
158 
159  else
160  {
161  // decrement count down to, but not below zero
162  *count = (*count-1) > 0 ? *count-1 : 0;
163  }
164  }
165 
166  return (bool)( *count >= threshold );
167 }

419 static void status_update (
420  pedometer_t* pedometerHandle,
421  uint32_t events,
422  bool suspend
423  )
424 {
425  bool
426  activity_stable = debounce_count(
427  pedometerHandle->config.bits.activity_dbcntm,
428  !(events & KEYNETIK_ACTIVITYCHANGED),
429  &pedometerHandle->private.activitycount,
430  pedometerHandle->config.activitycount_threshold
431  );
432 
433  activitylevel_t
434  activity = activity_stable ? keynetikActivityLevel : pedometerHandle->status.status.bits.activity;
435 
436  uint16_t
437  stepcount = uint32_to_uint16( keynetikStepCount );
438 
439  uint8_t
440  newstatus = ( activity << PEDOMETER_STATUS_ACTIVITY_BITNUM ) & PEDOMETER_STATUS_ACTIVITY_MASK;
441 
442  if ( 0 != ( ( newstatus ^ pedometerHandle->status.status.byte ) & PEDOMETER_STATUS_ACTIVITY_MASK ) )
443  {
444  newstatus |= PEDOMETER_STATUS_ACTCHG_MASK;
445  }
446 
447  if (
448  ( 0 != pedometerHandle->config.stepcoalesce )
449  && ( ( stepcount - pedometerHandle->private.stepchg_stepcount ) >= pedometerHandle->config.stepcoalesce )
450  )
451  {
452  pedometerHandle->private.stepchg_stepcount = stepcount;
453  newstatus |= PEDOMETER_STATUS_STEPCHG_MASK;
454  }
455 
456  if ( 0 != suspend )
457  {
458  newstatus |= PEDOMETER_STATUS_SUSPEND_MASK;
459  }
460 
461  if ( 0 != ( ( newstatus ^ pedometerHandle->status.status.byte ) & PEDOMETER_STATUS_SUSPEND_MASK ) )
462  {
463  newstatus |= PEDOMETER_STATUS_SUSPCHG_MASK;
464  }
465 
466  if ( 0 != ( newstatus & PEDOMETER_STATUS_CHG_MASK ) )
467  {
468  newstatus |= PEDOMETER_STATUS_MRGFLG_MASK;
469  }
470 
471  OSA_EnterCritical( kCriticalDisableInt );
472 
473  pedometerHandle->status.stepcount = uint32_to_uint16(keynetikStepCount);
474  pedometerHandle->status.distance = uint32_to_uint16(keynetikDistance);
475  pedometerHandle->status.speed = keynetikSpeed;
476  pedometerHandle->status.calories = uint32_to_uint16(keynetikCalories);
477  pedometerHandle->status.status.byte = newstatus;
478 
479  OSA_ExitCritical( kCriticalDisableInt );
