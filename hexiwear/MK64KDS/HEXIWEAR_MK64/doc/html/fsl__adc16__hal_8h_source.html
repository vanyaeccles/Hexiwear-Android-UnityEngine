281 typedef struct Adc16HwCmpConfig
282 {
283  bool hwCmpEnable;
284  bool hwCmpGreaterThanEnable;
285  /*
286  false - Configures less than the threshold. The outside and inside range are not inclusive.
287  The functionality is based on the values
288  placed in CV1 and CV2.
289  true - Configures greater than or equal to the threshold. The outside and inside
290  ranges are inclusive. The functionality is based on the values placed in
291  CV1 and CV2.
292  */
293  bool hwCmpRangeEnable;
294  /*
295  Configures the comparator function to check if the conversion result of the
296  input being monitored is either between or outside the range formed by
297  CV1 and CV2 and determined by the value of hwCmpGreaterThanEnable.
298 
299  false - Range function disabled. Only CV1 is compared.
300  true - Range function enabled. Both CV1 and CV2 are compared.
301  */
302  uint16_t cmpValue1;
303  uint16_t cmpValue2;
304 } adc16_hw_cmp_config_t;

324 typedef struct Adc16PgaConfig
325 {
326  bool pgaEnable;
327  bool runInNormalModeEnable;
328  adc16_pga_gain_t pgaGainMode;
330 #if FSL_FEATURE_ADC16_HAS_PGA_CHOPPING
331  bool pgaChoppingDisable;
332  /*
333  The PGA employs chopping to remove/reduce offset and 1/f noise and offers an
334  offset measurement configuration that aids the offset calibration.
335  */
336 #endif
337 #if FSL_FEATURE_ADC16_HAS_PGA_OFFSET_MEASUREMENT
338  bool runInOffsetMeasurementEnable;
339  /*
340  When this feature is enabled, the PGA disconnects itself from the external
341  inputs and auto-configures into offset measurement mode. With this bit set,
342  run the ADC in the recommended settings and enable the maximum hardware
343  averaging to get the PGA offset number. The output is the
344  (PGA offset * (64+1)) for the given PGA setting.
345  */
346 #endif
347 } adc16_pga_config_t;
