79 /*FUNCTION**********************************************************************
80  *
81  * Function Name : RTC_HAL_ConvertSecsToDatetime
82  * Description : converts time data from seconds to a datetime structure.
83  * This function will convert time data from seconds to a datetime structure.
84  *
85  *END*/
86 void RTC_HAL_ConvertSecsToDatetime(const uint32_t * seconds, rtc_datetime_t * datetime)
87 {
88  uint32_t x;
89  uint32_t Seconds, Days, Days_in_year;
90  const uint8_t *Days_in_month;
91 
93  Seconds = *seconds;
95  Days = Seconds / SECONDS_IN_A_DAY + 1;
97  Seconds = Seconds % SECONDS_IN_A_DAY;
99  datetime->hour = Seconds / SECONDS_IN_A_HOUR;
101  Seconds = Seconds % SECONDS_IN_A_HOUR;
103  datetime->minute = Seconds / SECONDS_IN_A_MIN;
105  datetime->second = Seconds % SECONDS_IN_A_MIN;
107  datetime->year = YEAR_RANGE_START;
108  Days_in_year = DAYS_IN_A_YEAR;
109 
110  while (Days > Days_in_year)
111  {
112  Days -= Days_in_year;
113  datetime->year++;
114  if (datetime->year & 3U)
115  {
116  Days_in_year = DAYS_IN_A_YEAR;
117  }
118  else
119  {
120  Days_in_year = DAYS_IN_A_LEAP_YEAR;
121  }
122  }
123 
124  if (datetime->year & 3U)
125  {
126  Days_in_month = ULY;
127  }
128  else
129  {
130  Days_in_month = LY;
131  }
132 
133  for (x = 1U; x <= 12U; x++)
134  {
135  if (Days <= (*(Days_in_month + x)))
136  {
137  datetime->month = x;
138  break;
139  }
140  else
141  {
142  Days -= (*(Days_in_month + x));
143  }
144  }
145 
146  datetime->day = Days;
147 }

149 /*FUNCTION**********************************************************************
150  *
151  * Function Name : RTC_HAL_IsDatetimeCorrectFormat
152  * Description : checks if the datetime is in correct format.
153  * This function will check if the given datetime is in the correct format.
154  *
155  *END*/
156 bool RTC_HAL_IsDatetimeCorrectFormat(const rtc_datetime_t * datetime)
157 {
158  bool result = false;
159 
161  if ((datetime->year < YEAR_RANGE_START) || (datetime->year > YEAR_RANGE_END) ||
162  (datetime->month > 12U) || (datetime->month < 1U) ||
163  (datetime->day > 31U) || (datetime->day < 1U) ||
164  (datetime->hour >= HOURS_IN_A_DAY) || (datetime->minute >= MINS_IN_A_HOUR) ||
165  (datetime->second >= SECONDS_IN_A_MIN))
166  {
168  result = false;
169  }
170  else
171  {
172  result = true;
173  }
174 
177  if ( result && (datetime->year & 3U))
178  {
180  if (ULY[datetime->month] < datetime->day)
181  {
183  result = false;
184  }
185  }
186  else
187  {
189  if (result && (LY[datetime->month] < datetime->day))
190  {
192  result = false;
193  }
194  }
195 
196  return result;
197 }

199 /*FUNCTION**********************************************************************
200  *
201  * Function Name : RTC_HAL_ConvertDatetimeToSecs
202  * Description : converts time data from datetime to seconds.
203  * This function will convert time data from datetime to seconds.
204  *
205  *END*/
206 void RTC_HAL_ConvertDatetimeToSecs(const rtc_datetime_t * datetime, uint32_t * seconds)
207 {
209  *seconds = (datetime->year - 1970U) * DAYS_IN_A_YEAR;
211  *seconds += ((datetime->year / 4) - (1970U / 4));
213  *seconds += MONTH_DAYS[datetime->month];
216  *seconds += (datetime->day - 1);
218  if ((!(datetime->year & 3U)) && (datetime->month <= 2U))
219  {
220  (*seconds)--;
221  }
222 
223  *seconds = ((*seconds) * SECONDS_IN_A_DAY) + (datetime->hour * SECONDS_IN_A_HOUR) +
224  (datetime->minute * SECONDS_IN_A_MIN) + datetime->second;
225 }
