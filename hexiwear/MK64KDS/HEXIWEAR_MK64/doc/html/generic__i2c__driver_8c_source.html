345 static statusI2C_t I2C_Reset( genericI2cHandle_t* self )
346 {
348  I2C_DRV_MasterDeinit( self->instance );
349 
350  uint32_t
351  tmtCnt = ( 2 * I2C_TIMEOUT )/10;
352 
355  uint32_t

361  PORT_Type*
362  portBase;
363 
365  switch ( self->instance )

370  portBase = g_portBase[1];
371  gpioBase = g_gpioBase[1];
372  sclPin = 0UL;

374  break;
375  }
376

378  case FSL_FS_I2C:

380  portBase = g_portBase[2];

382  sclPin = 10UL;
383  sdaPin = 11UL;
384  break;
385  }
386 
387  default:
388  {
389  return STATUS_I2C_INIT_ERROR;
390  break;
391  }
392  }
393 
395  PORT_HAL_SetMuxMode( portBase, sclPin, kPortMuxAsGpio);
396  PORT_HAL_SetMuxMode( portBase, sdaPin, kPortMuxAsGpio);
397 
399  GPIO_HAL_SetPinDir( gpioBase, sclPin, kGpioDigitalOutput);
400  GPIO_HAL_SetPinDir( gpioBase, sdaPin, kGpioDigitalOutput);
401 
403  PORT_HAL_SetOpenDrainCmd( portBase, sclPin, true );
404  PORT_HAL_SetOpenDrainCmd( portBase, sdaPin, true );
405 
408  while ( 0 == GPIO_HAL_ReadPinInput( gpioBase, sdaPin ) )
