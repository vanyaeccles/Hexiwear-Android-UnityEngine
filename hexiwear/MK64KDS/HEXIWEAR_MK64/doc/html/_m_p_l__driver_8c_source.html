501  ( mode > MPL_MODE_TEMPERATURE )
502  && ( mode < MPL_MODE_CURRENT )
503  )
504  {
505  return STATUS_MPL_INIT_ERROR;
506  }
507 
508  else if (
509  ( MPL_MODE_CURRENT != mode )
510  && ( selectedMode != mode )
511  )
512  {
513  // goto standby
514  status |= MPL_GotoStandby();
515  // set the mode
516  status |= MPL_SetMode(mode);
517  // goto active state
518  status |= MPL_SetActive();
519  }
520 
521  if ( STATUS_MPL_SUCCESS != status )
522  {
523  catch(8);
524  return STATUS_MPL_ERROR;
525  }
526 
527  else
528  {
533  switch ( selectedMode )
534  {
535  case MPL_MODE_PRESSURE:
536  case MPL_MODE_ALTITUDE: {
537  dataReadyFlag = 1 << MPL_PDR_SHIFT;
538  dataReadyRegAddr = OUT_P_MSB;
539  bytesToRead = 3;
540  readData = (uint8_t*)&(mplData.presAltDataMSB);
541  break;
542  }
543 
544  case MPL_MODE_TEMPERATURE: {
545  dataReadyFlag = 1 << MPL_TDR_SHIFT;
546  dataReadyRegAddr = OUT_T_MSB;
547  bytesToRead = 2;
548  readData = (uint8_t*)&(mplData.tempDataMSB);
549  break;
550  }
551 
552  default: {}
553  }
554 
555  statusI2C_t
556  status = I2C_ReadReg( &(self.protocol), REG_STATUS, &statVal);
557  if ( STATUS_I2C_SUCCESS != status )
558  {
559  catch(8);
560  return STATUS_MPL_PROTOCOL_ERROR;
561  }
562 
563  // check PDR/PTR bit; if it's not set, toggle OST
564  else if ( 0 == ( statVal & dataReadyFlag ) )
565  {
566  // toggle the OST bit, causing the sensor to immediately take another reading
567  MPL_ToggleOneShot(self);
568  }
569 
570  // wait for the data acquisition time
571 // OSA_TimeDelay(selectedDelay);
572 // OSA_TimeDelay( 10 );
573 
574  int16_t
575  counter = 0;
576 
577  // wait for PDR/PTR bit, which indicates that we have new data
578  while (1)
579  {
580  status = I2C_ReadReg( &(self.protocol), REG_STATUS, &statVal);
581  if ( STATUS_I2C_SUCCESS != status )
582  {
583  return STATUS_MPL_PROTOCOL_ERROR;
584  }
585 
586  else
587  {
588  if ( 0 == ( statVal & dataReadyFlag ) )
589  {
590  // error out after more than 0.5s for a read op
591  if ( ++counter > 5 )
592  {
593  return STATUS_MPL_TIMEOUT;
594  }
595  }
596 
597  else
598  {
599  // continue with the program
600  break;
601  }
602  }
603  }
604

610  i2cProtocol = self.protocol;
611

613  i2cStatus = I2C_DRV_MasterReceiveDataBlocking (
614  i2cProtocol.instance, // I2C module being used
615  &(i2cProtocol.device), // device-specific settings
616

627  {
628  catch(8);

630  }
631 
632  else
633  {
634 
639  switch ( selectedMode )
640  {
641  // pressure value is a Q18.2 right-aligned number in [Pa]
642  case MPL_MODE_PRESSURE:
643  // altitude value is a Q16.4 right-aligned number in [m]
644  case MPL_MODE_ALTITUDE: {
645  // save pressure/altitude
646  uint32_t
647  foo = (
648  // MSB
649  ( mplData.presAltDataMSB << 16 )
650  // CSB
651  | ( mplData.presAltDataCSB << 8 )
652  // LSB
653  | ( mplData.presAltDataLSB )
654  // discard first 4 bits
655  ) >> 4;
656 
657  // pack it into the first two integers in the output array
658  memcpy( (void*)sensorData, (const void*)&foo, sizeof(foo) );
659 
660  break;
661  }
662 
663  // temperature value is a Q8.4 right-aligned number in [C]
664  case MPL_MODE_TEMPERATURE: {
665  // save temperature MSB LSB
666  *sensorData = (int16_t)( ( mplData.tempDataMSB << 4 ) | ( mplData.tempDataLSB >> 4 ) );
667 
668  break;
669  }
670 
671  default: {}
672  }
673  }
674 
675  return STATUS_MPL_SUCCESS;
676  }
