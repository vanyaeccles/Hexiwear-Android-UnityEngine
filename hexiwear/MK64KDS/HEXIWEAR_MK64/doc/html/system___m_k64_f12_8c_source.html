310 void SystemCoreClockUpdate (void) {
311  uint32_t MCGOUTClock;
312  uint16_t Divider;
313 
314  if ((MCG->C1 & MCG_C1_CLKS_MASK) == 0x00U) {
316  if ((MCG->C6 & MCG_C6_PLLS_MASK) == 0x00U) {
318  if ((MCG->C1 & MCG_C1_IREFS_MASK) == 0x00U) {
320  switch (MCG->C7 & MCG_C7_OSCSEL_MASK) {
321  case 0x00U:
322  MCGOUTClock = CPU_XTAL_CLK_HZ;
323  break;
324  case 0x01U:
325  MCGOUTClock = CPU_XTAL32k_CLK_HZ;
326  break;
327  case 0x02U:
328  default:
329  MCGOUTClock = CPU_INT_IRC_CLK_HZ;
330  break;
331  }
332  if (((MCG->C2 & MCG_C2_RANGE_MASK) != 0x00U) && ((MCG->C7 & MCG_C7_OSCSEL_MASK) != 0x01U)) {
333  switch (MCG->C1 & MCG_C1_FRDIV_MASK) {
334  case 0x38U:
335  Divider = 1536U;
336  break;
337  case 0x30U:
338  Divider = 1280U;
339  break;
340  default:
341  Divider = (uint16_t)(32LU << ((MCG->C1 & MCG_C1_FRDIV_MASK) >> MCG_C1_FRDIV_SHIFT));
342  break;
343  }
344  } else {/* ((MCG->C2 & MCG_C2_RANGE_MASK) != 0x00U) */
345  Divider = (uint16_t)(1LU << ((MCG->C1 & MCG_C1_FRDIV_MASK) >> MCG_C1_FRDIV_SHIFT));
346  }
347  MCGOUTClock = (MCGOUTClock / Divider);
348  } else { /* (!((MCG->C1 & MCG_C1_IREFS_MASK) == 0x00U)) */
349  MCGOUTClock = CPU_INT_SLOW_CLK_HZ;
350  } /* (!((MCG->C1 & MCG_C1_IREFS_MASK) == 0x00U)) */
352  switch (MCG->C4 & (MCG_C4_DMX32_MASK | MCG_C4_DRST_DRS_MASK)) {
353  case 0x00U:
354  MCGOUTClock *= 640U;
355  break;
356  case 0x20U:
357  MCGOUTClock *= 1280U;
358  break;
359  case 0x40U:
360  MCGOUTClock *= 1920U;
361  break;
362  case 0x60U:
363  MCGOUTClock *= 2560U;
364  break;
365  case 0x80U:
366  MCGOUTClock *= 732U;
367  break;
368  case 0xA0U:
369  MCGOUTClock *= 1464U;
370  break;
371  case 0xC0U:
372  MCGOUTClock *= 2197U;
373  break;
374  case 0xE0U:
375  MCGOUTClock *= 2929U;
376  break;
377  default:
378  break;
379  }
380  } else { /* (!((MCG->C6 & MCG_C6_PLLS_MASK) == 0x00U)) */
382  Divider = (((uint16_t)MCG->C5 & MCG_C5_PRDIV0_MASK) + 0x01U);
383  MCGOUTClock = (uint32_t)(CPU_XTAL_CLK_HZ / Divider);
384  Divider = (((uint16_t)MCG->C6 & MCG_C6_VDIV0_MASK) + 24U);
385  MCGOUTClock *= Divider;
386  } /* (!((MCG->C6 & MCG_C6_PLLS_MASK) == 0x00U)) */
387  } else if ((MCG->C1 & MCG_C1_CLKS_MASK) == 0x40U) {
389  if ((MCG->C2 & MCG_C2_IRCS_MASK) == 0x00U) {
390  MCGOUTClock = CPU_INT_SLOW_CLK_HZ;
391  } else { /* (!((MCG->C2 & MCG_C2_IRCS_MASK) == 0x00U)) */
392  Divider = (uint16_t)(0x01LU << ((MCG->SC & MCG_SC_FCRDIV_MASK) >> MCG_SC_FCRDIV_SHIFT));
393  MCGOUTClock = (uint32_t) (CPU_INT_FAST_CLK_HZ / Divider);
394  } /* (!((MCG->C2 & MCG_C2_IRCS_MASK) == 0x00U)) */
395  } else if ((MCG->C1 & MCG_C1_CLKS_MASK) == 0x80U) {
397  switch (MCG->C7 & MCG_C7_OSCSEL_MASK) {
398  case 0x00U:
399  MCGOUTClock = CPU_XTAL_CLK_HZ;
400  break;
401  case 0x01U:
402  MCGOUTClock = CPU_XTAL32k_CLK_HZ;
403  break;
404  case 0x02U:
405  default:
406  MCGOUTClock = CPU_INT_IRC_CLK_HZ;
407  break;
408  }
409  } else { /* (!((MCG->C1 & MCG_C1_CLKS_MASK) == 0x80U)) */
411  return;
412  } /* (!((MCG->C1 & MCG_C1_CLKS_MASK) == 0x80U)) */
413  SystemCoreClock = (MCGOUTClock / (0x01U + ((SIM->CLKDIV1 & SIM_CLKDIV1_OUTDIV1_MASK) >> SIM_CLKDIV1_OUTDIV1_SHIFT)));
414 }
