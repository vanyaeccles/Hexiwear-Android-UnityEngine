234  );

236  }

362 static void heartRate_FindHR(
363  float32_t* sigFFT,
364  float32_t minFreq,
365  float32_t maxFreq,
366  uint8_t* heartRateVal
367  )
368 {
369  uint16_t
370  minIdx = (uint16_t)floor( minFreq * fft_currSize / HR_SAMPLE_RATE ),
371  maxIdx = (uint16_t)ceil ( maxFreq * fft_currSize / HR_SAMPLE_RATE );
372 
373  uint16_t
374  currPeak;
375 
376  uint32_t
377  peaks[ MAXIM_FIFO_DEPTH/2 ] = { 0 };
378 
384  uint32_t
385  peakIdx = 0;
386 
387  for ( uint16_t i = minIdx; i <= maxIdx; i++ )
388  {
389  if ( 0 != heartRate_IsPeak( sigFFT, i ) )
390  {
391  peaks[ peakIdx ] = i;
392  peakIdx += 1;
393  }
394  }
395 
396  // find the biggest among all the found peaks
397  if ( peakIdx > 0 )
398  {
399  currPeak = peaks[0];
400  for ( uint16_t i = 0; i < peakIdx; i++ )
401  {
402  if ( sigFFT [ currPeak ] < sigFFT[ peaks[i] ] )
403  {
404  currPeak = peaks[i];
405  }
406  }
407 
408  // convert to BPM
409  *heartRateVal = (uint8_t)( 60 * ( ( (float32_t)currPeak * HR_SAMPLE_RATE ) / fft_currSize ) );
410  }
411  else
412  {
413  *heartRateVal = 0xFF;
414  }
415 }
