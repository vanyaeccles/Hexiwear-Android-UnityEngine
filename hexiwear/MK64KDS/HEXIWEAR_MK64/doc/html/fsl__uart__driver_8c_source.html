60 /*FUNCTION**********************************************************************
61  *
62  * Function Name : UART_DRV_Init
63  * Description : This function initializes a UART instance for operation.
64  * This function will initialize the run-time state structure to keep track of
65  * the on-going transfers, ungate the clock to the UART module, initialize the
66  * module to user defined settings and default settings, configure the IRQ state
67  * structure and enable the module-level interrupt to the core, and enable the
68  * UART module transmitter and receiver.
69  * The following is an example of how to set up the uart_state_t and the
70  * uart_user_config_t parameters and how to call the UART_DRV_Init function by
71  * passing in these parameters:
72  * uart_user_config_t uartConfig;
73  * uartConfig.baudRate = 9600;
74  * uartConfig.bitCountPerChar = kUart8BitsPerChar;
75  * uartConfig.parityMode = kUartParityDisabled;
76  * uartConfig.stopBitCount = kUartOneStopBit;
77  * uart_state_t uartState;
78  * UART_DRV_Init(instance, &uartState, &uartConfig);
79  *
80  *END*/
81 uart_status_t UART_DRV_Init(uint32_t instance, uart_state_t * uartStatePtr,
82  const uart_user_config_t * uartUserConfig)
83 {
84  assert(uartStatePtr && uartUserConfig);
85  assert(g_uartBase[instance]);
86  assert(instance < UART_INSTANCE_COUNT);
87 
88  UART_Type * base = g_uartBase[instance];
89  uint32_t uartSourceClock;
90 
92  if (g_uartStatePtr[instance])
93  {
94  return kStatus_UART_Initialized;
95  }
96 
98  memset(uartStatePtr, 0, sizeof(uart_state_t));
99 
101  g_uartStatePtr[instance] = uartStatePtr;
102 
104  CLOCK_SYS_EnableUartClock(instance);
105 
107  UART_HAL_Init(base);
108 
110  OSA_SemaCreate(&uartStatePtr->txIrqSync, 0);
111  OSA_SemaCreate(&uartStatePtr->rxIrqSync, 0);
112 
114  uartSourceClock = CLOCK_SYS_GetUartFreq(instance);
115 
117  UART_HAL_SetBaudRate(base, uartSourceClock, uartUserConfig->baudRate);
118  UART_HAL_SetBitCountPerChar(base, uartUserConfig->bitCountPerChar);
119  UART_HAL_SetParityMode(base, uartUserConfig->parityMode);
120 #if FSL_FEATURE_UART_HAS_STOP_BIT_CONFIG_SUPPORT
121  UART_HAL_SetStopBitCount(base, uartUserConfig->stopBitCount);
122 #endif
123 
124 #if FSL_FEATURE_UART_HAS_FIFO
125  uint8_t fifoSize;
127  fifoSize = UART_HAL_GetTxFifoSize(base);
129  uartStatePtr->txFifoEntryCount = (fifoSize == 0 ? 1 : 0x1 << (fifoSize + 1));
130 
138  if (uartStatePtr->txFifoEntryCount > 1)
139  {
140  UART_HAL_SetTxFifoWatermark(base, (uartStatePtr->txFifoEntryCount >> 1U));
141  }
142  else
143  {
144  UART_HAL_SetTxFifoWatermark(base, 0);
145  }
146 
155  UART_HAL_SetRxFifoWatermark(base, 1);
156 
158  UART_HAL_SetTxFifoCmd(base, true);
159  UART_HAL_SetRxFifoCmd(base, true);
160  UART_HAL_FlushTxFifo(base);
161  UART_HAL_FlushRxFifo(base);
162 #else
163 
167  uartStatePtr->txFifoEntryCount = 1;
168 #endif
169 
171  INT_SYS_EnableIRQ(g_uartRxTxIrqId[instance]);
172 
174  UART_HAL_EnableTransmitter(base);
175  UART_HAL_EnableReceiver(base);
176 
177  return kStatus_UART_Success;
178 }

180 /*FUNCTION**********************************************************************
181  *
182  * Function Name : UART_DRV_Deinit
183  * Description : This function shuts down the UART by disabling interrupts and
184  * the transmitter/receiver.
185  * This function disables the UART interrupts, disables the transmitter and
186  * receiver, and flushes the FIFOs (for modules that support FIFOs).
187  *
188  *END*/
189 uart_status_t UART_DRV_Deinit(uint32_t instance)
190 {
191  assert(instance < UART_INSTANCE_COUNT);
192  assert(g_uartBase[instance]);
193 
195  if ((!g_uartStatePtr[instance]) || (!CLOCK_SYS_GetUartGateCmd(instance)))
196  {
197  return kStatus_UART_Fail;
198  }
199 
200  UART_Type * base = g_uartBase[instance];
201  uart_state_t * uartState = (uart_state_t *)g_uartStatePtr[instance];
202 
205 #if FSL_FEATURE_UART_HAS_FIFO
206 
207  while(UART_HAL_GetTxDatawordCountInFifo(base) != 0) { }
208 #endif
209 
210  while(!(UART_BRD_S1_TC(base))) { }
211 
213  INT_SYS_DisableIRQ(g_uartRxTxIrqId[instance]);
214 
216  UART_HAL_DisableTransmitter(base);
217  UART_HAL_DisableReceiver(base);
218 
220  OSA_SemaDestroy(&uartState->txIrqSync);
221  OSA_SemaDestroy(&uartState->rxIrqSync);
222 
223 #if FSL_FEATURE_UART_HAS_FIFO
224 
225  UART_HAL_SetTxFifoCmd(base, false);
226  UART_HAL_SetRxFifoCmd(base, false);
227  UART_HAL_FlushTxFifo(base);
228  UART_HAL_FlushRxFifo(base);
229 #endif
230 
232  g_uartStatePtr[instance] = NULL;
233 
235  CLOCK_SYS_DisableUartClock(instance);
236 
237  return kStatus_UART_Success;
238 }

292 /*FUNCTION**********************************************************************
293  *
294  * Function Name : UART_DRV_SendDataBlocking
295  * Description : This function sends (transmits) data out through the UART
296  * module using a blocking method.
297  * A blocking (also known as synchronous) function means that the function does
298  * not return until the transmit is complete. This blocking function is used to
299  * send data through the UART port.
300  *
301  *END*/
302 uart_status_t UART_DRV_SendDataBlocking(uint32_t instance,
303  const uint8_t * txBuff,
304  uint32_t txSize,
305  uint32_t timeout)
306 {
307  assert(txBuff);
308  assert(instance < UART_INSTANCE_COUNT);
309 
310  uart_state_t * uartState = (uart_state_t *)g_uartStatePtr[instance];
311  UART_Type * base = g_uartBase[instance];
312  uart_status_t retVal = kStatus_UART_Success;
313  osa_status_t syncStatus;
314 
316  uartState->isTxBlocking = true;
317 
319  retVal = UART_DRV_StartSendData(instance, txBuff, txSize);
320 
321  if (retVal == kStatus_UART_Success)
322  {
324  do
325  {
326  syncStatus = OSA_SemaWait(&uartState->txIrqSync, timeout);
327  }while(syncStatus == kStatus_OSA_Idle);
328 
329  if (syncStatus != kStatus_OSA_Success)
330  {
332  UART_BWR_C2_TIE(base, 0U);
333 
335  uartState->isTxBusy = false;
336 
337  retVal = kStatus_UART_Timeout;
338  }
339 
340 #if FSL_FEATURE_UART_HAS_FIFO
341 
342  while(UART_HAL_GetTxDatawordCountInFifo(base) != 0) { }
343 #endif
344  }
345 
346  return retVal;
347 }

349 /*FUNCTION**********************************************************************
350  *
351  * Function Name : UART_DRV_SendData
352  * Description : This function sends (transmits) data through the UART module
353  * using a non-blocking method.
354  * A non-blocking (also known as asynchronous) function means that the function
355  * returns immediately after initiating the transmit function. The application
356  * has to get the transmit status to see when the transmit is complete. In
357  * other words, after calling non-blocking (asynchronous) send function, the
358  * application must get the transmit status to check if transmit is completed
359  * or not. The asynchronous method of transmitting and receiving allows the UART
360  * to perform a full duplex operation (simultaneously transmit and receive).
361  *
362  *END*/
363 uart_status_t UART_DRV_SendData(uint32_t instance,
364  const uint8_t * txBuff,
365  uint32_t txSize)
366 {
367  assert(txBuff);
368  assert(instance < UART_INSTANCE_COUNT);
369 
370  uart_status_t retVal = kStatus_UART_Success;
371  uart_state_t * uartState = (uart_state_t *)g_uartStatePtr[instance];
372 
374  uartState->isTxBlocking = false;
375 
377  retVal = UART_DRV_StartSendData(instance, txBuff, txSize);
378 
379  return retVal;
380 }

382 /*FUNCTION**********************************************************************
383  *
384  * Function Name : UART_DRV_GetTransmitStatus
385  * Description : This function returns whether the previous UART transmit has
386  * finished.
387  * When performing an async transmit, the user can call this function to
388  * ascertain the state of the current transmission: in progress (or busy) or
389  * complete (success). In addition, if the transmission is still in progress,
390  * the user can obtain the number of words that have been currently transferred.
391  *
392  *END*/
393 uart_status_t UART_DRV_GetTransmitStatus(uint32_t instance, uint32_t * bytesRemaining)
394 {
395  assert(instance < UART_INSTANCE_COUNT);
396 
397  uart_state_t * uartState = (uart_state_t *)g_uartStatePtr[instance];
398  uart_status_t retVal = kStatus_UART_Success;
399  uint32_t txSize = uartState->txSize;
400 
404  if (bytesRemaining)
405  {
406  *bytesRemaining = txSize;
407  }
408 
409  if (txSize)
410  {
411  retVal = kStatus_UART_TxBusy;
412  }
413 
414 #if FSL_FEATURE_UART_HAS_FIFO
415  UART_Type * base = g_uartBase[instance];
416 
417  if (UART_HAL_GetTxDatawordCountInFifo(base))
418  {
419  retVal = kStatus_UART_TxBusy;
420  }
421 #endif
422 
423  return retVal;
424 }

592 /*FUNCTION**********************************************************************
593  *
594  * Function Name : UART_DRV_IRQHandler
595  * Description : Interrupt handler for UART.
596  * This handler uses the buffers stored in the uart_state_t structs to transfer
597  * data. This is not a public API as it is called whenever an interrupt occurs.
598  *
599  *END*/
600 void UART_DRV_IRQHandler(uint32_t instance)
601 {
602  uart_state_t * uartState = (uart_state_t *)g_uartStatePtr[instance];
603  UART_Type * base = g_uartBase[instance];
604 
606  if ((!uartState->isTxBusy) && (!uartState->isRxBusy))
607  {
608  return;
609  }
610 
613  if((UART_BRD_C2_RIE(base)) && (UART_BRD_S1_RDRF(base)))
614  {
615 #if FSL_FEATURE_UART_HAS_FIFO
616 
617  while(UART_HAL_GetRxDatawordCountInFifo(base))
618  {
619 #endif
620 
621  UART_HAL_Getchar(base, uartState->rxBuff);
622 
624  if (uartState->rxCallback != NULL)
625  {
626  uartState->rxCallback(instance, uartState);
627  }
628  else
629  {
630  ++uartState->rxBuff;
631  --uartState->rxSize;
632 
634  if (uartState->rxSize == 0U)
635  {
636  UART_DRV_CompleteReceiveData(instance);
637  #if FSL_FEATURE_UART_HAS_FIFO
638  break;
639  #endif
640  }
641  }
642 #if FSL_FEATURE_UART_HAS_FIFO
643  }
644 #endif
645  }
646 
649  if((UART_BRD_C2_TIE(base)) && (UART_BRD_S1_TDRE(base)))
650  {
652  if (uartState->txSize)
653  {
654  uint8_t emptyEntryCountInFifo;
655 #if FSL_FEATURE_UART_HAS_FIFO
656  emptyEntryCountInFifo = uartState->txFifoEntryCount -
657  UART_HAL_GetTxDatawordCountInFifo(base);
658 #else
659  emptyEntryCountInFifo = uartState->txFifoEntryCount;
660 #endif
661  while(emptyEntryCountInFifo--)
662  {
664  UART_HAL_Putchar(base, *(uartState->txBuff));
665 
667  if (uartState->txCallback != NULL)
668  {
671  uartState->txCallback(instance, uartState);
672  }
673  else
674  {
675  ++uartState->txBuff;
676  --uartState->txSize;
677  }
678 
680  if (uartState->txSize == 0U)
681  {
682  UART_DRV_CompleteSendData(instance);
683  break;
684  }
685  }
686  }
687  }
688 
690  if (UART_HAL_GetStatusFlag(base, kUartRxOverrun))
691  {
693  UART_HAL_ClearStatusFlag(base, kUartRxOverrun);
694  }
695 }

697 /*FUNCTION**********************************************************************
698  *
699  * Function Name : UART_DRV_CompleteSendData
700  * Description : Finish up a transmit by completing the process of sending
701  * data and disabling the interrupt.
702  * This is not a public API as it is called from other driver functions.
703  *
704  *END*/
705 static void UART_DRV_CompleteSendData(uint32_t instance)
706 {
707  assert(instance < UART_INSTANCE_COUNT);
708 
709  UART_Type * base = g_uartBase[instance];
710  uart_state_t * uartState = (uart_state_t *)g_uartStatePtr[instance];
711 
713  UART_BWR_C2_TIE(base, 0U);
714 
716  if (uartState->isTxBlocking)
717  {
718  OSA_SemaPost(&uartState->txIrqSync);
719  }
720 
722  uartState->isTxBusy = false;
723 }

725 /*FUNCTION**********************************************************************
726  *
727  * Function Name : UART_DRV_StartSendData
728  * Description : Initiate (start) a transmit by beginning the process of
729  * sending data and enabling the interrupt.
730  * This is not a public API as it is called from other driver functions.
731  *
732  *END*/
733 static uart_status_t UART_DRV_StartSendData(uint32_t instance,
734  const uint8_t * txBuff,
735  uint32_t txSize)
736 {
737  assert(instance < UART_INSTANCE_COUNT);
738 
739  UART_Type * base = g_uartBase[instance];
740  uart_state_t * uartState = (uart_state_t *)g_uartStatePtr[instance];
741 
744  if (uartState->isTxBusy)
745  {
746  return kStatus_UART_TxBusy;
747  }
748 
749  if (txSize == 0U)
750  {
751  return kStatus_UART_NoDataToDeal;
752  }
753 
755  uartState->txBuff = txBuff;
756  uartState->txSize = txSize;
757  uartState->isTxBusy = true;
758 
763  UART_BWR_C2_TIE(base, 1U);
764 
765  return kStatus_UART_Success;
766 }

768 /*FUNCTION**********************************************************************

771  * Description : Finish up a receive by completing the process of receiving
