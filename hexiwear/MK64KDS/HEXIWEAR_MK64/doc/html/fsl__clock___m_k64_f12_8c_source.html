61 /*FUNCTION**********************************************************************
62  *
63  * Function Name : CLOCK_SYS_FllStableDelay
64  * Description : This funtion is used to delay for FLL stable.
65  * According to datasheet, every time the FLL reference source or reference
66  * divider is changed, trim value is changed, DMX32 bit is changed, DRS bits
67  * are changed, or changing from FLL disabled (BLPE, BLPI) to FLL enabled
68  * (FEI, FEE, FBE, FBI), there should be 1ms delay for FLL stable. Please
69  * check datasheet for t(fll_aquire).
70  *
71  *END*/
72 static void CLOCK_SYS_FllStableDelay(void)
73 {
74  uint32_t coreClk = CLOCK_SYS_GetCoreClockFreq();
75 
76  coreClk /= 3000U;
77 
78  // Delay 1 ms.
79  while (coreClk--)
80  {
81  __asm ("nop");
82  }
83 }

104 /*FUNCTION**********************************************************************
105  *
106  * Function Name : CLOCK_SYS_SetConfiguration
107  * Description : This funtion sets the system to target configuration, it
108  * only sets the clock modules registers for clock mode change, but not send
109  * notifications to drivers.
110  *
111  *END*/
112 clock_manager_error_code_t CLOCK_SYS_SetConfiguration(clock_manager_user_config_t const* config)
113 {
114  assert(NULL != config);
115 
117  CLOCK_HAL_SetOutDiv(SIM, 0U, 1U, 2U, 4U);
118 
120  CLOCK_SYS_SetMcgMode(&config->mcgConfig, CLOCK_SYS_FllStableDelay);
121 
123  CLOCK_SYS_SetSimConfigration(&config->simConfig);
124 
126  CLOCK_SYS_SetOscerConfigration(0, &config->oscerConfig);
127  SystemCoreClock = CLOCK_SYS_GetCoreClockFreq();
128 
129  return kClockManagerSuccess;
130 }

132 clock_manager_error_code_t CLOCK_SYS_GetFreq(clock_names_t clockName,
133  uint32_t *frequency)
134 {
135  clock_manager_error_code_t returnCode = kClockManagerSuccess;
136 
137  switch (clockName)
138  {
139  case kCoreClock:
140  case kSystemClock:
141  *frequency = CLOCK_SYS_GetCoreClockFreq();
142  break;
143  case kPlatformClock:
144  *frequency = CLOCK_SYS_GetSystemClockFreq();
145  break;
146  case kBusClock:
147  *frequency = CLOCK_SYS_GetBusClockFreq();
148  break;
149  case kFlexBusClock:
150  *frequency = CLOCK_SYS_GetFlexbusFreq();
151  break;
152  case kFlashClock:
153  *frequency = CLOCK_SYS_GetFlashClockFreq();
154  break;
155  case kOsc32kClock:
156  *frequency = CLOCK_SYS_GetExternalRefClock32kFreq();
157  break;
158  case kOsc0ErClock:
159  *frequency = CLOCK_SYS_GetOsc0ExternalRefClockFreq();
160  break;
161  case kRtcoutClock:
162  *frequency = CLOCK_SYS_GetRtcOutFreq();
163  break;
164  case kMcgFfClock:
165  *frequency = CLOCK_SYS_GetFixedFreqClockFreq();
166  break;
167  case kMcgFllClock:
168  *frequency = CLOCK_HAL_GetFllClk(MCG);
169  break;
170  case kMcgPll0Clock:
171  *frequency = CLOCK_HAL_GetPll0Clk(MCG);
172  break;
173  case kMcgOutClock:
174  *frequency = CLOCK_HAL_GetOutClk(MCG);
175  break;
176  case kMcgIrClock:
177  *frequency = CLOCK_HAL_GetInternalRefClk(MCG);
178  break;
179  case kIrc48mClock:
180  *frequency = CPU_INTERNAL_IRC_48M;
181  break;
182  case kLpoClock:
183  *frequency = CLOCK_SYS_GetLpoClockFreq();
184  break;
185  case kSystickClock:
186  *frequency = CLOCK_SYS_GetSystickFreq();
187  break;
188  default:
189  *frequency = 0U;
190  returnCode = kClockManagerNoSuchClockName;
191  break;
192  }
193 
194  return returnCode;
195 }
