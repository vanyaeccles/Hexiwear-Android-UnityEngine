61 /*FUNCTION**********************************************************************
62  *
63  * Function Name : OSA_SemaWait
64  * Description : This function checks the semaphore's counting value, if it is
65  * positive, decreases it and returns kStatus_OSA_Success, otherwise, timeout
66  * will be used for wait. The parameter timeout indicates how long should wait
67  * in milliseconds. Pass OSA_WAIT_FOREVER to wait indefinitely, pass 0 will
68  * return kStatus_OSA_Timeout immediately if semaphore is not positive.
69  * This function returns kStatus_OSA_Success if the semaphore is received, returns
70  * kStatus_OSA_Timeout if the semaphore is not received within the specified
71  * 'timeout', returns kStatus_OSA_Error if any errors occur during waiting.
72  *
73  *END*/
74 osa_status_t OSA_SemaWait(semaphore_t *pSem, uint32_t timeout)
75 {
76  uint32_t timeoutTicks;
77  assert(pSem);
78 
80  if (timeout == OSA_WAIT_FOREVER)
81  {
82  timeoutTicks = portMAX_DELAY;
83  }
84  else
85  {
86  timeoutTicks = MSEC_TO_TICK(timeout);
87  }
88 
89  if (xSemaphoreTake(*pSem, timeoutTicks)==pdFALSE)
90  {
91  return kStatus_OSA_Timeout;
92  }
93  else
94  {
95  return kStatus_OSA_Success;
96  }
97 }

192  *
193  * Function Name : OSA_MutexLock
194  * Description : This function checks the mutex's status, if it is unlocked,
195  * lock it and returns kStatus_OSA_Success, otherwise, timeout will be used for
196  * wait. The parameter timeout indicates how long should wait in milliseconds.
197  * Pass OSA_WAIT_FOREVER to wait indefinitely, pass 0 will return the value
198  * kStatus_OSA_Timeout immediately if mutex is locked.
199  * This function returns kStatus_OSA_Success if the mutex is obtained, returns
200  * kStatus_OSA_Timeout if the mutex is not obtained within the specified
201  * 'timeout', returns kStatus_OSA_Error if any errors occur during waiting.
202  *
203  *END*/
204 osa_status_t OSA_MutexLock(mutex_t *pMutex, uint32_t timeout)
205 {
206  uint32_t timeoutTicks;
207 
208  assert(pMutex);
209 
211  if (xSemaphoreGetMutexHolder(*pMutex) == xTaskGetCurrentTaskHandle())
212  {
213  return kStatus_OSA_Error;
214  }
215 
217  if (timeout == OSA_WAIT_FOREVER)
218  {
219  timeoutTicks = portMAX_DELAY;
220  }
221  else
222  {
223  timeoutTicks = MSEC_TO_TICK(timeout);
224  }
225 
226  if (xSemaphoreTake(*pMutex, timeoutTicks)==pdFALSE)
227  {
228  return kStatus_OSA_Timeout;
229  }
230  else
231  {
232  return kStatus_OSA_Success;
233  }
234 }

303 /*FUNCTION**********************************************************************
304  *
305  * Function Name : OSA_EventWait
306  * Description : This function checks the event's status, if it meets the wait
307  * condition, return kStatus_OSA_Success, otherwise, timeout will be used for
308  * wait. The parameter timeout indicates how long should wait in milliseconds.
309  * Pass OSA_WAIT_FOREVER to wait indefinitely, pass 0 will return the value
310  * kStatus_OSA_Timeout immediately if wait condition is not met. The event flags
311  * will be cleared if the event is auto clear mode. Flags that wakeup waiting
312  * task could be obtained from the parameter setFlags.
313  * This function returns kStatus_OSA_Success if wait condition is met, returns
314  * kStatus_OSA_Timeout if wait condition is not met within the specified
315  * 'timeout', returns kStatus_OSA_Error if any errors occur during waiting.
316  *
317  *END*/
318 osa_status_t OSA_EventWait(event_t *pEvent,
319  event_flags_t flagsToWait,
320  bool waitAll,
321  uint32_t timeout,
322  event_flags_t *setFlags)
323 {
324  assert(pEvent);
325  BaseType_t clearMode;
326  uint32_t timeoutTicks;
327  event_flags_t flagsSave;
328 
330  if (timeout == OSA_WAIT_FOREVER)
331  {
332  timeoutTicks = portMAX_DELAY;
333  }
334  else
335  {
336  timeoutTicks = timeout/portTICK_PERIOD_MS;
337  }
338 
339  clearMode = (kEventAutoClear == pEvent->clearMode) ? pdTRUE: pdFALSE;
340 
341  flagsSave = xEventGroupWaitBits(pEvent->eventHandler,
342  flagsToWait,
343  clearMode,
344  (BaseType_t)waitAll,
345  timeoutTicks);
346 
347  *setFlags = flagsSave & flagsToWait;
348 
349  if (*setFlags)
350  {
351  return kStatus_OSA_Success;
352  }
353  else
354  {
355  return kStatus_OSA_Timeout;
356  }
357 }
