108 void *pvPortMalloc( size_t xWantedSize )
109 {
110 void *pvReturn = NULL;
111 static uint8_t *pucAlignedHeap = NULL;
112 
114  #if portBYTE_ALIGNMENT != 1
115  if( xWantedSize & portBYTE_ALIGNMENT_MASK )
116  {
118  xWantedSize += ( portBYTE_ALIGNMENT - ( xWantedSize & portBYTE_ALIGNMENT_MASK ) );
119  }
120  #endif
121 
122  vTaskSuspendAll();
123  {
124  if( pucAlignedHeap == NULL )
125  {
127  pucAlignedHeap = ( uint8_t * ) ( ( ( portPOINTER_SIZE_TYPE ) &ucHeap[ portBYTE_ALIGNMENT ] ) & ( ( portPOINTER_SIZE_TYPE ) ~portBYTE_ALIGNMENT_MASK ) );
128  }
129 
131  if( ( ( xNextFreeByte + xWantedSize ) < configADJUSTED_HEAP_SIZE ) &&
132  ( ( xNextFreeByte + xWantedSize ) > xNextFreeByte ) )
133  {
136  pvReturn = pucAlignedHeap + xNextFreeByte;
137  xNextFreeByte += xWantedSize;
138  }
139 
140  traceMALLOC( pvReturn, xWantedSize );
141  }
142  (void)xTaskResumeAll();
143 
144  #if( configUSE_MALLOC_FAILED_HOOK == 1 )
145  {
146  if( pvReturn == NULL )
147  {
148  extern void vApplicationMallocFailedHook( void );
149  vApplicationMallocFailedHook();
150  }
151  }
152  #endif
153 
154  return pvReturn;
155 }
