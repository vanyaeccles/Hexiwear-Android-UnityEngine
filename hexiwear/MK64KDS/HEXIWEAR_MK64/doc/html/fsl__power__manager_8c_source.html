112 /*FUNCTION**********************************************************************
113  *
114  * Function Name : POWER_SYS_Init
115  * Description : Initializes the Power manager for operation.
116  * This function initializes the Power manager and its run-time state structure.
117  * Reference to an array of Power mode configuration structures has to be passed
118  * as parameter along with parameter specifying its size. At least one power mode
119  * configuration is required. Optionally, reference to array of predefined
120  * call-backs can be passed with its size parameter.
121  * For details about call-backs refer to the power_manager_callback_user_config_t.
122  * As Power manager stores only references to array of these structures they have
123  * to exist while Power manager is used.
124  * It is expected that prior POWER_SYS_Init() call the write-once protection
125  * register was configured appropriately allowing to enter all required low power
126  * modes.
127  * The following is an example of how to set up two power modes and three
128  * call-backs and initialize the Power manager with structures containing their settings.
129  * The example shows two possible ways how where the configuration structures can be stored
130  * (ROM or RAM) although it is expected that they will be placed rather in the read-only
131  * memory to save the RAM space. (Note: In the example it is assumed that the programmed chip
132  * doesn't support any optional power options described in the power_manager_user_config_t)
133  *
134  *END*/
135 power_manager_error_code_t POWER_SYS_Init(power_manager_user_config_t const ** powerConfigsPtr,
136  uint8_t configsNumber,
137  power_manager_callback_user_config_t ** callbacksPtr,
138  uint8_t callbacksNumber)
139 {
141  if ((powerConfigsPtr == NULL) || (configsNumber == 0U))
142  {
143  return kPowerManagerError;
144  }
146  POWER_SYS_LOCK_INIT();
148  memset(&gPowerManagerState, 0, sizeof(power_manager_state_t));
150  gPowerManagerState.configs = powerConfigsPtr;
151  gPowerManagerState.configsNumber = configsNumber;
153  if (callbacksPtr != NULL)
154  {
155  gPowerManagerState.staticCallbacks = callbacksPtr;
156  gPowerManagerState.staticCallbacksNumber = callbacksNumber;
158  gPowerManagerState.errorCallbackIndex = callbacksNumber;
159  }
161 #if FSL_FEATURE_SIM_HAS_SCGC_LLWU
162  SIM_HAL_EnableClock(SIM,kSimClockGateLlwu0);
163 #endif
164  return kPowerManagerSuccess;
165 }

180 /*FUNCTION**********************************************************************
181  *
182  * Function Name : POWER_SYS_SetMode
183  * Description : Configures the power mode.
184  *
185  * This function switches to one of the defined power modes. Requested mode number is passed
186  * as an input parameter. This function notifies all registered callback functions before
187  * the mode change (using kPowerManagerCallbackBefore set as callback type parameter),
188  * sets specific power options defined in the power mode configuration and enters the specified
189  * mode. In case of run modes (for example, Run, Very low power run, or High speed run), this function
190  * also invokes all registered callbacks after the mode change (using kPowerManagerCallbackAfter).
191  * In case of sleep or deep sleep modes, if the requested mode is not exited through
192  * a reset, these notifications are sent after the core wakes up.
193  * Callbacks are invoked in the following order: All registered callbacks are notified
194  * ordered by index in the callbacks array (see callbacksPtr parameter of POWER_SYS_Init()).
195  * The same order is used for before and after switch notifications.
196  * The notifications before the power mode switch can be used to obtain confirmation about
197  * the change from registered callbacks. If any registered callback denies the power
198  * mode change, further execution of this function depends on mode change policy: the mode
199  * change is either forced (kPowerManagerPolicyForcible) or exited (kPowerManagerPolicyAgreement).
200  * When mode change is forced, the result of the before switch notifications are ignored. If
201  * agreement is required, if any callback returns an error code then further notifications
202  * before switch notifications are cancelled and all already notified callbacks are re-invoked
203  * with kPowerManagerCallbackAfter set as callback type parameter. The index of the callback
204  * which returned error code during pre-switch notifications is stored (any error codes during
205  * callbacks re-invocation are ignored) and POWER_SYS_GetErrorCallback() can be used to get it.
206  * Regardless of the policies, if any callback returned an error code, an error code denoting in which phase
207  * the error occurred is returned when POWER_SYS_SetMode() exits.
208  * It is possible to enter any mode supported by the processor. Refer to the chip reference manual
209  * for list of available power modes. If it is necessary to switch into intermediate power mode prior to
210  * entering requested mode (for example, when switching from Run into Very low power wait through Very low
211  * power run mode), then the intermediate mode is entered without invoking the callback mechanism.
212  *
213  *END*/
214 power_manager_error_code_t POWER_SYS_SetMode(uint8_t powerModeIndex, power_manager_policy_t policy)
215 {
216  power_manager_user_config_t const * configPtr; /*
217  Local pointer to the requested user-defined power mode configuration */
218  power_manager_modes_t mode;
219  smc_power_mode_config_t halModeConfig;
220  uint8_t currentStaticCallback = 0U;
221  power_manager_error_code_t returnCode = kPowerManagerSuccess;
222  power_manager_error_code_t clockCheckRetCode;
223  power_manager_notify_struct_t notifyStruct; /* Callback notification structure */
224  power_manager_callback_user_config_t * callbackConfig;
227  gPowerManagerState.errorCallbackIndex = gPowerManagerState.staticCallbacksNumber;
228 
229  POWER_SYS_LOCK();
231  if (powerModeIndex >= gPowerManagerState.configsNumber)
232  {
233  POWER_SYS_UNLOCK();
234  return kPowerManagerErrorOutOfRange;
235  }
236 
238  configPtr = gPowerManagerState.configs[powerModeIndex];
239  mode = configPtr->mode;
240  notifyStruct.policy = policy;
241  notifyStruct.targetPowerConfigIndex = powerModeIndex;
242  notifyStruct.targetPowerConfigPtr = configPtr;
243 
245  if ((mode == kPowerManagerVlpr) || (mode == kPowerManagerVlpw) || (mode == kPowerManagerVlps))
246  {
247  if (!SMC_HAL_GetProtection(SMC, kAllowPowerModeVlp))
248  {
249  POWER_SYS_UNLOCK();
250  return kPowerManagerError;
251  }
252  }
253 #if FSL_FEATURE_SMC_HAS_LOW_LEAKAGE_STOP_MODE
254 #if FSL_FEATURE_SMC_HAS_STOP_SUBMODE0
255  else if ((mode >= kPowerManagerLls) && (mode < kPowerManagerVlls0))
256 #else
257  else if ((mode >= kPowerManagerLls) && (mode < kPowerManagerVlls1))
258 #endif
259  {
260  if (!SMC_HAL_GetProtection(SMC, kAllowPowerModeLls))
261  {
262  POWER_SYS_UNLOCK();
263  return kPowerManagerError;
264  }
265  }
266 #endif
267 #if FSL_FEATURE_SMC_HAS_STOP_SUBMODE0
268  else if (mode >= kPowerManagerVlls0)
269 #else
270  else if (mode >= kPowerManagerVlls1)
271 #endif
272  {
273  if (!SMC_HAL_GetProtection(SMC, kAllowPowerModeVlls))
274  {
275  POWER_SYS_UNLOCK();
276  return kPowerManagerError;
277  }
278  }
279 
280  notifyStruct.notifyType = kPowerManagerNotifyBefore;
282  for (currentStaticCallback = 0U; currentStaticCallback < gPowerManagerState.staticCallbacksNumber; currentStaticCallback++)
283  {
284  callbackConfig = (gPowerManagerState.staticCallbacks[currentStaticCallback]);
286  if ( callbackConfig != NULL ){
287  /* ...notify only those which asked to be called before the power mode change */
288  if (((uint32_t)callbackConfig->callbackType) & kPowerManagerCallbackBefore)
289  {
291  if (callbackConfig->callback(&notifyStruct, callbackConfig->callbackData) != kPowerManagerSuccess)
292  {
293  returnCode = kPowerManagerErrorNotificationBefore;
294  gPowerManagerState.errorCallbackIndex = currentStaticCallback;
296  if (policy != kPowerManagerPolicyForcible)
297  {
298  break;
299  }
300  }
301  }
302  }
303  }
304 
305  POWER_SYS_UNLOCK();
306 
308  clockCheckRetCode = POWER_SYS_CheckClocks(mode);
309 
313  if ( ((policy == kPowerManagerPolicyForcible) || (returnCode == kPowerManagerSuccess)) && (clockCheckRetCode == kPowerManagerSuccess))
314  {
315 #if FSL_FEATURE_SMC_HAS_LPWUI
316  halModeConfig.lpwuiOptionValue = configPtr->lowPowerWakeUpOnInterruptValue;
317 #endif
318 
320  switch (mode) {
321 #if FSL_FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE
322 
323  case kPowerManagerHsrun:
324  halModeConfig.powerModeName = kPowerModeHsrun;
325  break;
326 #endif
327 
328  case kPowerManagerRun:
329  halModeConfig.powerModeName = kPowerModeRun;
330  break;
332  case kPowerManagerVlpr:
333  halModeConfig.powerModeName = kPowerModeVlpr;
334  break;
336  case kPowerManagerWait:
337  halModeConfig.powerModeName = kPowerModeWait;
338  break;
340  case kPowerManagerVlpw:
341  halModeConfig.powerModeName = kPowerModeVlpw;
342  break;
343 #if FSL_FEATURE_SMC_HAS_LOW_LEAKAGE_STOP_MODE
344 
345  case kPowerManagerLls:
346  halModeConfig.powerModeName = kPowerModeLls;
347  break;
348 #if FSL_FEATURE_SMC_HAS_LLS_SUBMODE
349  case kPowerManagerLls2:
350  halModeConfig.powerModeName = kPowerModeLls;
351  halModeConfig.stopSubMode = kSmcStopSub2;
352  break;
353  case kPowerManagerLls3:
354  halModeConfig.powerModeName = kPowerModeLls;
355  halModeConfig.stopSubMode = kSmcStopSub3;
356  break;
357 #endif
358 #endif
359 
360 #if FSL_FEATURE_SMC_HAS_STOP_SUBMODE0
361  case kPowerManagerVlls0:
362  halModeConfig.powerModeName = kPowerModeVlls;
363  halModeConfig.stopSubMode = kSmcStopSub0;
364 #if FSL_FEATURE_SMC_HAS_PORPO
365 
366  halModeConfig.porOptionValue = configPtr->powerOnResetDetectionValue;
367 #endif
368  break;
369 #endif
370  case kPowerManagerVlls1:
371  halModeConfig.powerModeName = kPowerModeVlls;
372  halModeConfig.stopSubMode = kSmcStopSub1;
373  break;
374 #if FSL_FEATURE_SMC_HAS_STOP_SUBMODE2
375  case kPowerManagerVlls2:
376  halModeConfig.powerModeName = kPowerModeVlls;
377  halModeConfig.stopSubMode = kSmcStopSub2;
378 #if FSL_FEATURE_SMC_HAS_RAM2_POWER_OPTION
379 
380  halModeConfig.ram2OptionValue = configPtr->RAM2PartitionValue;
381 #endif
382  break;
383 #endif
384  case kPowerManagerVlls3:
385  halModeConfig.powerModeName = kPowerModeVlls;
386  halModeConfig.stopSubMode = kSmcStopSub3;
387  break;
388 #if FSL_FEATURE_SMC_HAS_PSTOPO
389 
390  case kPowerManagerPstop1:
391  halModeConfig.powerModeName = kPowerModeStop;
392  halModeConfig.pstopOptionValue = kSmcPstopStop1;
393  break;
394  case kPowerManagerPstop2:
395  halModeConfig.powerModeName = kPowerModeStop;
396  halModeConfig.pstopOptionValue = kSmcPstopStop2;
397  break;
398 #endif
399 
400  case kPowerManagerStop:
401  halModeConfig.powerModeName = kPowerModeStop;
402 #if FSL_FEATURE_SMC_HAS_PSTOPO
403  halModeConfig.pstopOptionValue = kSmcPstopStop;
404 #endif
405  break;
407  case kPowerManagerVlps:
408  halModeConfig.powerModeName = kPowerModeVlps;
409  break;
410  default:
411  return kPowerManagerErrorSwitch;
412  }
413 
414 #if FSL_FEATURE_SMC_HAS_LPOPO
415 #if FSL_FEATURE_SMC_HAS_LOW_LEAKAGE_STOP_MODE
416  if ((mode >= kPowerManagerLls) && (mode <= kPowerManagerVlls3))
417 #else
418 #if FSL_FEATURE_SMC_HAS_STOP_SUBMODE0
419  if ((mode >= kPowerManagerVlls0) && (mode <= kPowerManagerVlls3))
420 #else
421  if ((mode >= kPowerManagerVlls1) && (mode <= kPowerManagerVlls3))
422 #endif
423 #endif
424  {
426  halModeConfig.lpoOptionValue = configPtr->lowPowerOscillatorValue;
427  }
428 #endif
429 
431  if (mode >= kPowerManagerWait)
432  {
433  if (configPtr->sleepOnExitValue)
434  {
436  SCB->SCR |= SCB_SCR_SLEEPONEXIT_Msk;
437  }
438  else
439  {
441  SCB->SCR &= ~(SCB_SCR_SLEEPONEXIT_Msk);
442  }
443  }
444 
446  if (SMC_HAL_SetMode(SMC, &halModeConfig) != kSmcHalSuccess)
447  {
448  returnCode = kPowerManagerErrorSwitch;
449  }
450 
452  if (mode == kPowerManagerRun)
453  {
454  returnCode = POWER_SYS_WaitForRunStatus();
455 
456  if(returnCode != kPowerManagerSuccess)
457  {
458  return returnCode;
459  }
460  }
461  else if (mode == kPowerManagerVlpr)
462  {
463  returnCode = POWER_SYS_WaitForVlprStatus();
464 
465  if(returnCode != kPowerManagerSuccess)
466  {
467  return returnCode;
468  }
469  }
470 
473  POWER_SYS_LOCK();
475  gPowerManagerState.currentConfig = powerModeIndex;
476  notifyStruct.notifyType = kPowerManagerNotifyAfter;
478  for (currentStaticCallback = 0U; currentStaticCallback < gPowerManagerState.staticCallbacksNumber; currentStaticCallback++)
479  {
480  callbackConfig = (gPowerManagerState.staticCallbacks[currentStaticCallback]);
482  if ( callbackConfig != NULL ){
483  /* ...notify only those which asked to be called after the power mode change */
484  if (((uint32_t)callbackConfig->callbackType) & kPowerManagerCallbackAfter)
485  {
487  if (callbackConfig->callback(&notifyStruct, callbackConfig->callbackData) != kPowerManagerSuccess)
488  {
489  returnCode = kPowerManagerErrorNotificationAfter;
490  gPowerManagerState.errorCallbackIndex = currentStaticCallback;
491 
492  }
493  }
494  }
495  }
496  POWER_SYS_UNLOCK();
497  }
498  else
499  {
503  if((policy == kPowerManagerPolicyForcible) || (returnCode == kPowerManagerSuccess))
504  {
505  returnCode = clockCheckRetCode;
506  }
507 
508  POWER_SYS_LOCK();
509 
510  notifyStruct.notifyType = kPowerManagerNotifyRecover;
511  while(currentStaticCallback--)
512  {
513  callbackConfig = (gPowerManagerState.staticCallbacks[currentStaticCallback]);
515  if ( callbackConfig != NULL ){
516  if (((uint32_t)callbackConfig->callbackType) & kPowerManagerCallbackBefore)
517  {
518  callbackConfig->callback(&notifyStruct, callbackConfig->callbackData);
519  }
520  }
521  }
522  POWER_SYS_UNLOCK();
523 
524  }
525 
526  return returnCode;
527 }

847 {

849  mcglite_mode_t mcgMode = CLOCK_HAL_GetMode(MCG);
850 #else
851  mcg_modes_t mcgMode = CLOCK_HAL_GetMcgMode(MCG);
852 
854  switch(mode)
855  {
856 #if FSL_FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE
857  case kPowerManagerHsrun:
858 #endif
859  case kPowerManagerRun:
860  case kPowerManagerWait:
862  break;
863  default:
865 #if FSL_FEATURE_MCG_HAS_EXTERNAL_CLOCK_MONITOR
866  if( CLOCK_HAL_IsOsc0MonitorEnabled(MCG) )
867  {
868  return kPowerManagerErrorClock;
869  }
870 #endif
871 #if FSL_FEATURE_MCG_HAS_RTC_32K
872  if( CLOCK_HAL_IsRtcOscMonitorEnabled(MCG) )
873  {
874  return kPowerManagerErrorClock;
875  }
876 #endif
877 #if FSL_FEATURE_MCG_USE_PLLREFSEL
878  if( CLOCK_HAL_IsOsc1MonitorEnabled(MCG) )
879  {
880  return kPowerManagerErrorClock;
881  }
882 #endif
883  break;
884  }
885 #endif
886 
888  if( (mode!=kPowerManagerVlpr) && (mode!=kPowerManagerVlpw) )
889  {
890  return kPowerManagerSuccess;
891  }
892 
893 #if POWER_VLPR_MCG_LITE
894  switch(mcgMode)
895  {
896  case kMcgliteModeLirc8M:
897  case kMcgliteModeLirc2M:
898  case kMcgliteModeExt:
900  break;
901  default:
902  return kPowerManagerErrorClock;
903  }
904 
905  if(CLOCK_SYS_GetFlashClockFreq() > POWER_VLPR_MAX_FLASH_BLPE_CLK)
906  {
907  return kPowerManagerErrorClock;
908  }
909 #else
910  switch(mcgMode)
911  {
912  case kMcgModeBLPI:

915  {
916  return kPowerManagerErrorClock;
917  }

919  {

921  }

924  if(CLOCK_SYS_GetFlashClockFreq() > POWER_VLPR_MAX_FLASH_BLPE_CLK)
925  {
926  return kPowerManagerErrorClock;
927  }
928  break;
929  default:
930  return kPowerManagerErrorClock;
931  }
932 #endif
933  if(CLOCK_SYS_GetCoreClockFreq() > POWER_VLPR_MAX_CLK)
934  {
935  return kPowerManagerErrorClock;
936  }
937 
938  if(CLOCK_SYS_GetBusClockFreq() > POWER_VLPR_MAX_CLK)

941  }
942

944  {

946  }
947 
948  return kPowerManagerSuccess;
