113 /*FUNCTION**********************************************************************
114  *
115  * Function Name : DSPI_DRV_EdmaSlaveInit
116  * Description : Initialize a DSPI slave instance.
117  * Un-gate DSPI's clock, setup the default value and initializes the required
118  * resources.
119  *
120  *END*/
121 dspi_status_t DSPI_DRV_EdmaSlaveInit(uint32_t instance,
122  dspi_edma_slave_state_t * dspiState,
123  const dspi_edma_slave_user_config_t * slaveConfig)
124 {
125  dspi_status_t result = kStatus_DSPI_Success;
126  dma_request_source_t dspiTxEdmaRequest = kDmaRequestMux0Disable;
127  dma_request_source_t dspiRxEdmaRequest = kDmaRequestMux0Disable;
128  SPI_Type *base = g_dspiBase[instance];
129 
131  if ((dspiState == NULL) || (slaveConfig == NULL))
132  {
133  return kStatus_DSPI_InvalidParameter;
134  }
135 
137  if (g_dspiStatePtr[instance])
138  {
139  return kStatus_DSPI_Initialized;
140  }
141 
143  if (slaveConfig->dataConfig.bitsPerFrame > 16)
144  {
145  return kStatus_DSPI_OutOfRange;
146  }
147 
149  memset(dspiState, 0, sizeof(* dspiState));
150 
152  dspiState->status = kStatus_DSPI_Success;
153  dspiState->errorCount = 0;
154  dspiState->dummyPattern = slaveConfig->dummyPattern;
155  dspiState->remainingSendByteCount = 0;
156  dspiState->remainingReceiveByteCount = 0;
157  dspiState->isTransferInProgress = false;
158  dspiState->extraReceiveByte = 0;
159 
160  if (kStatus_OSA_Success != OSA_EventCreate(&dspiState->event, kEventAutoClear))
161  {
163  dspiState->status = kStatus_DSPI_Error;
164  return kStatus_DSPI_Error;
165  }
166 
168  dspiState->bitsPerFrame = slaveConfig->dataConfig.bitsPerFrame;
169 
171  CLOCK_SYS_EnableSpiClock(instance);
172 
174  DSPI_HAL_Init(base);
175 
177  DSPI_HAL_SetMasterSlaveMode(base, kDspiSlave);
178 
180  result = DSPI_HAL_SetDataFormat(base, kDspiCtar0, &slaveConfig->dataConfig);
181 
183  DSPI_HAL_SetFifoCmd(base, true, true);
184 
186  DSPI_HAL_SetFlushFifoCmd(base, true, true);
187 
188  switch (instance)
189  {
190  case 0:
192 #if FSL_FEATURE_DSPI_HAS_SEPARATE_DMA_RX_TX_REQn(0)
193  dspiRxEdmaRequest = kDmaRequestMux0SPI0Rx;
194  dspiTxEdmaRequest = kDmaRequestMux0SPI0Tx;
195 #else
196  dspiRxEdmaRequest = kDmaRequestMux0SPI0;
197  dspiTxEdmaRequest = kDmaRequestMux0Disable;
198 #endif
199  break;
200 #if (SPI_INSTANCE_COUNT > 1)
201  case 1:
203 #if FSL_FEATURE_DSPI_HAS_SEPARATE_DMA_RX_TX_REQn(1)
204  dspiRxEdmaRequest = kDmaRequestMux0SPI1Rx;
205  dspiTxEdmaRequest = kDmaRequestMux0SPI1Tx;
206 #else
207  dspiRxEdmaRequest = kDmaRequestMux0SPI1;
208  dspiTxEdmaRequest = kDmaRequestMux0Disable;
209 #endif
210  break;
211 #endif
212 #if (SPI_INSTANCE_COUNT > 2)
213  case 2:
215 #if FSL_FEATURE_DSPI_HAS_SEPARATE_DMA_RX_TX_REQn(2)
216  dspiRxEdmaRequest = kDmaRequestMux0SPI2Rx;
217  dspiTxEdmaRequest = kDmaRequestMux0SPI2Tx;
218 #else
219  dspiRxEdmaRequest = kDmaRequestMux0SPI2;
220  dspiTxEdmaRequest = kDmaRequestMux0Disable;
221 #endif
222  break;
223 #endif
224  default :
225  return kStatus_DSPI_InvalidInstanceNumber;
226  }
227 
229  if (kEDMAInvalidChannel == EDMA_DRV_RequestChannel(kEDMAAnyChannel,
230  dspiRxEdmaRequest,
231  &dspiState->edmaRxChannel))
232  {
233  dspiState->status = kStatus_DSPI_Error;
234  return kStatus_DSPI_DMAChannelInvalid;
235  }
236 
238  if (kEDMAInvalidChannel == EDMA_DRV_RequestChannel(kEDMAAnyChannel,
239  dspiTxEdmaRequest,
240  &dspiState->edmaTxChannel))
241  {
242  dspiState->status = kStatus_DSPI_Error;
243  return kStatus_DSPI_DMAChannelInvalid;
244  }
245 
247  g_dspiStatePtr[instance] = dspiState;
248 
250  INT_SYS_EnableIRQ(g_dspiIrqId[instance]);
251 
253  DSPI_HAL_Enable(base);
254 
255  return result;
256 }

258 /*FUNCTION**********************************************************************
259  *
260  * Function Name : DSPI_DRV_EdmaSlaveDeinit
261  * Description : Shutdown a DSPI instance.
262  * Resets the DSPI peripheral, disables the interrupt to the core, and gates its clock.
263  *
264  *END*/
265 dspi_status_t DSPI_DRV_EdmaSlaveDeinit(uint32_t instance)
266 {
267  SPI_Type *base = g_dspiBase[instance];
268  dspi_edma_slave_state_t * dspiState = (dspi_edma_slave_state_t *)g_dspiStatePtr[instance];
269 
271  assert(instance < SPI_INSTANCE_COUNT);
272 
273  if (!dspiState)
274  {
275  return kStatus_DSPI_NonInit;
276  }
277 
279  INT_SYS_DisableIRQ(g_dspiIrqId[instance]);
280 
282  DSPI_HAL_StopTransfer(base);
283 
293  while((DSPI_HAL_GetStatusFlag(base, kDspiTxAndRxStatus))) { }
294 
296  DSPI_HAL_Init(base);
297 
299  CLOCK_SYS_DisableSpiClock(instance);
300 
302  OSA_EventDestroy(&dspiState->event);
303 
304  EDMA_DRV_ReleaseChannel(&dspiState->edmaTxChannel);
305  EDMA_DRV_ReleaseChannel(&dspiState->edmaRxChannel);
306 
307  dspiState->status = kStatus_DSPI_NonInit;
308 
310  g_dspiStatePtr[instance] = NULL;
311 
312  return kStatus_DSPI_Success;
313 }

315 /*FUNCTION**********************************************************************
316  *
317  * Function Name : DSPI_DRV_EdmaSlaveStartTransfer
318  * Description : Starts transfer data on SPI bus using eDMA and non-blocking call
319  * Start DSPI transfering, update transmit/receive information into slave state structure
320  *
321  *END*/
322 static void DSPI_DRV_EdmaSlaveStartTransfer(uint32_t instance,
323  const uint8_t *sendBuffer,
324  uint8_t *receiveBuffer,
325  uint32_t transferByteCount)
326 {
327  DMA_Type * dmaBaseAddr;
328  uint32_t dmaChannel;
329  uint32_t majorIteration;
330  SPI_Type *base = g_dspiBase[instance];
331  dspi_edma_slave_state_t * dspiState = (dspi_edma_slave_state_t *)g_dspiStatePtr[instance];
332  uint8_t nBytes = dspiState->bitsPerFrame / 8;
333  uint32_t txTransferByteCnt = 0;
334  uint32_t rxTransferByteCnt = 0;
335  uint32_t sendData = 0;
336 
338  if (dspiState->bitsPerFrame % 8 != 0)
339  {
340  nBytes ++;
341  }
342 
344  s_dataToSend = dspiState->dummyPattern;
345 
347  DSPI_HAL_StopTransfer(base);
348 
350  DSPI_HAL_PresetTransferCount(base, 0);
351 
353  DSPI_HAL_SetFlushFifoCmd(base, true, true);
354 
356  DSPI_HAL_ClearStatusFlag(base, kDspiTxComplete);
357  DSPI_HAL_ClearStatusFlag(base, kDspiTxAndRxStatus);
358  DSPI_HAL_ClearStatusFlag(base, kDspiEndOfQueue);
359  DSPI_HAL_ClearStatusFlag(base, kDspiTxFifoUnderflow);
360  DSPI_HAL_ClearStatusFlag(base, kDspiTxFifoFillRequest);
361  DSPI_HAL_ClearStatusFlag(base, kDspiRxFifoOverflow);
362  DSPI_HAL_ClearStatusFlag(base, kDspiRxFifoDrainRequest);
363 
365  DSPI_HAL_StartTransfer(base);
367  while(!DSPI_HAL_GetStatusFlag(base,kDspiTxFifoFillRequest))
368  {}
370  if (sendBuffer)
371  {
372  if (nBytes == 1)
373  {
374  sendData = *sendBuffer;
375  sendBuffer ++;
376  }
377  else
378  {
379  sendData = *sendBuffer;
380  sendBuffer ++;
381  sendData |= (uint32_t)(*sendBuffer) << 8;
382  sendBuffer++;
383  }
384  }
385  else
386  {
387  sendData = s_dataToSend;
388  }
389  DSPI_HAL_WriteDataSlavemode(base, sendData);
391  DSPI_HAL_ClearStatusFlag(base, kDspiTxFifoFillRequest);
392 
394  if ((nBytes > 1) && ((transferByteCount % nBytes) != 0))
395  {
396  rxTransferByteCnt = (transferByteCount/nBytes) * nBytes;
397  dspiState->extraReceiveByte = transferByteCount - rxTransferByteCnt;
399  dspiState->remainingSendByteCount = (transferByteCount/nBytes + 1) * nBytes;
400  }
401  else
402  {
403  dspiState->extraReceiveByte = 0;
404  rxTransferByteCnt = transferByteCount;
406  dspiState->remainingSendByteCount = transferByteCount;
407  }
409  txTransferByteCnt = rxTransferByteCnt;
410 
412  dspiState->remainingReceiveByteCount = rxTransferByteCnt;
413 
415  dspiState->sendBuffer = sendBuffer;
416  if (txTransferByteCnt)
417  {
419  dmaBaseAddr = VIRTUAL_CHN_TO_EDMA_MODULE_REGBASE(dspiState->edmaTxChannel.channel);
420  dmaChannel = VIRTUAL_CHN_TO_EDMA_CHN(dspiState->edmaTxChannel.channel);
421  EDMA_HAL_ClearDoneStatusFlag(dmaBaseAddr, (edma_channel_indicator_t)dmaChannel);
422  EDMA_HAL_HTCDClearReg(dmaBaseAddr, dmaChannel);
423 
424  if (sendBuffer)
425  {
427  EDMA_HAL_HTCDSetSrcAddr(dmaBaseAddr, dmaChannel,(uint32_t)(sendBuffer));
428  }
429  else
430  {
432  EDMA_HAL_HTCDSetSrcAddr(dmaBaseAddr, dmaChannel,(uint32_t)(&s_dataToSend));
433  }
434 
436  EDMA_HAL_HTCDSetSrcLastAdjust(dmaBaseAddr, dmaChannel, 0);
437 
438 
440  EDMA_HAL_HTCDSetDestAddr(dmaBaseAddr, dmaChannel,
441  DSPI_HAL_GetSlavePushrRegAddr(base));
442 
444  EDMA_HAL_HTCDSetDestOffset(dmaBaseAddr, dmaChannel, 0);
445 
447  EDMA_HAL_HTCDSetDestLastAdjust(dmaBaseAddr, dmaChannel, 0);
448 
449  /* * The source and destination attributes (bit size) depends on bits/frame setting */
450  if (dspiState->bitsPerFrame <= 8)
451  {
452  if (sendBuffer)
453  {
457  EDMA_HAL_HTCDSetSrcOffset(dmaBaseAddr, dmaChannel, 1);
458  }
459  else
460  {
462  EDMA_HAL_HTCDSetSrcOffset(dmaBaseAddr, dmaChannel, 0);
463  }
464 
466  EDMA_HAL_HTCDSetAttribute(dmaBaseAddr, dmaChannel,
467  kEDMAModuloDisable, kEDMAModuloDisable,
468  kEDMATransferSize_1Bytes, kEDMATransferSize_1Bytes);
469 
471  EDMA_HAL_HTCDSetNbytes(dmaBaseAddr, dmaChannel, 1);
472 
474  majorIteration = txTransferByteCnt;
475  }
477  else
478  {
479  if (sendBuffer)
480  {
484  EDMA_HAL_HTCDSetSrcOffset(dmaBaseAddr, dmaChannel, 2);
485  }
486  else
487  {
489  EDMA_HAL_HTCDSetSrcOffset(dmaBaseAddr, dmaChannel, 0);
490  }
491 
493  EDMA_HAL_HTCDSetAttribute(dmaBaseAddr, dmaChannel,
494  kEDMAModuloDisable, kEDMAModuloDisable,
495  kEDMATransferSize_2Bytes, kEDMATransferSize_2Bytes);
496 
498  EDMA_HAL_HTCDSetNbytes(dmaBaseAddr, dmaChannel, 2);
499 
501  majorIteration = txTransferByteCnt/2;
502  }
503 
505  EDMA_HAL_HTCDSetChannelMinorLink(dmaBaseAddr, dmaChannel, 0, false);
506  EDMA_HAL_HTCDSetMajorCount(dmaBaseAddr, dmaChannel, majorIteration);
507 
510  EDMA_HAL_SetDmaRequestCmd(dmaBaseAddr, (edma_channel_indicator_t)dmaChannel, true);
511 
513  EDMA_HAL_HTCDSetIntCmd(dmaBaseAddr, dmaChannel, true);
514  }
515  else
516  {
518  DSPI_HAL_SetTxFifoFillDmaIntMode(base, kDspiGenerateIntReq, false);
519  }
520 
522  dspiState->receiveBuffer = receiveBuffer;
523  if (rxTransferByteCnt)
524  {
526  dmaBaseAddr = VIRTUAL_CHN_TO_EDMA_MODULE_REGBASE(dspiState->edmaRxChannel.channel);
527  dmaChannel = VIRTUAL_CHN_TO_EDMA_CHN(dspiState->edmaRxChannel.channel);
528  EDMA_HAL_ClearDoneStatusFlag(dmaBaseAddr, (edma_channel_indicator_t)dmaChannel);
529  EDMA_HAL_HTCDClearReg(dmaBaseAddr, dmaChannel);
530 
532  EDMA_HAL_HTCDSetSrcAddr(dmaBaseAddr, dmaChannel,
533  DSPI_HAL_GetPoprRegAddr(base));
534 
536  EDMA_HAL_HTCDSetSrcOffset(dmaBaseAddr, dmaChannel, 0);
537 
539  EDMA_HAL_HTCDSetSrcLastAdjust(dmaBaseAddr, dmaChannel, 0);
540 
541  if (receiveBuffer)
542  {
544  EDMA_HAL_HTCDSetDestAddr(dmaBaseAddr, dmaChannel, (uint32_t)(receiveBuffer));
545  }
546  else
547  {
549  EDMA_HAL_HTCDSetDestAddr(dmaBaseAddr, dmaChannel, (uint32_t)(&s_rxBuffIfNull));
550  }
551 
553  EDMA_HAL_HTCDSetDestLastAdjust(dmaBaseAddr, dmaChannel, 0);
554 
556  if (dspiState->bitsPerFrame <= 8)
557  {
558  if (receiveBuffer)
559  {
561  EDMA_HAL_HTCDSetDestOffset(dmaBaseAddr, dmaChannel, 1);
562  }
563  else
564  {
566  EDMA_HAL_HTCDSetDestOffset(dmaBaseAddr, dmaChannel, 0);
567  }
568 
570  EDMA_HAL_HTCDSetAttribute(dmaBaseAddr, dmaChannel,
571  kEDMAModuloDisable, kEDMAModuloDisable,
572  kEDMATransferSize_1Bytes, kEDMATransferSize_1Bytes);
573 
575  EDMA_HAL_HTCDSetNbytes(dmaBaseAddr, dmaChannel, 1);
576 
578  majorIteration = rxTransferByteCnt;
579  }
580  else
582  {
583  if (receiveBuffer)
584  {
586  EDMA_HAL_HTCDSetDestOffset(dmaBaseAddr, dmaChannel, 2);
587  }
588  else
589  {
591  EDMA_HAL_HTCDSetDestOffset(dmaBaseAddr, dmaChannel, 0);
592  }
593 
595  EDMA_HAL_HTCDSetAttribute(dmaBaseAddr, dmaChannel,
596  kEDMAModuloDisable, kEDMAModuloDisable,
597  kEDMATransferSize_2Bytes, kEDMATransferSize_2Bytes);
598 
600  EDMA_HAL_HTCDSetNbytes(dmaBaseAddr, dmaChannel, 2);
601 
603  majorIteration = rxTransferByteCnt/2;
604  }
605 
609  if (FSL_FEATURE_DSPI_HAS_SEPARATE_DMA_RX_TX_REQn(instance))
610  {
612  EDMA_HAL_HTCDSetChannelMinorLink(dmaBaseAddr, dmaChannel, 0, false);
613  }
614  else
615  {
617  EDMA_HAL_HTCDSetChannelMinorLink(
618  dmaBaseAddr,
619  dmaChannel,
620  VIRTUAL_CHN_TO_EDMA_CHN(dspiState->edmaTxChannel.channel),
621  true);
625  EDMA_HAL_HTCDSetChannelMajorLink(
626  dmaBaseAddr,
627  dmaChannel,
628  VIRTUAL_CHN_TO_EDMA_CHN(dspiState->edmaTxChannel.channel),
629  true);
630  }
631  EDMA_HAL_HTCDSetMajorCount(dmaBaseAddr, dmaChannel, majorIteration);
632 
636  EDMA_HAL_SetDmaRequestCmd(dmaBaseAddr, (edma_channel_indicator_t)dmaChannel, true);
637 
644  EDMA_DRV_InstallCallback(&dspiState->edmaRxChannel,
645  DSPI_DRV_EdmaRxCallback,(void *)instance);
646 
648  EDMA_HAL_HTCDSetIntCmd(dmaBaseAddr, dmaChannel, true);
649  }
650  else if (dspiState->extraReceiveByte)
651  {
653  DSPI_HAL_ClearStatusFlag(base, kDspiTxComplete);
654  DSPI_HAL_SetIntMode(base, kDspiTxComplete, true);
655  }
656  else
657  {
659  DSPI_HAL_SetRxFifoDrainDmaIntMode(base, kDspiGenerateIntReq, false);
660  }
661 
662  if(txTransferByteCnt)
663  {
667  if (dspiState->remainingSendByteCount <= (g_dspiFifoSize[instance] * nBytes))
668  {
669  DSPI_HAL_ClearStatusFlag(base, kDspiTxComplete);
670  DSPI_HAL_SetIntMode(base, kDspiTxComplete, true);
671  }
672  else
673  {
674  DSPI_HAL_SetIntMode(base, kDspiTxComplete, false);
675  }
676  dmaBaseAddr = VIRTUAL_CHN_TO_EDMA_MODULE_REGBASE(dspiState->edmaTxChannel.channel);
677  dmaChannel = VIRTUAL_CHN_TO_EDMA_CHN(dspiState->edmaTxChannel.channel);
678 
682  if (FSL_FEATURE_DSPI_HAS_SEPARATE_DMA_RX_TX_REQn(instance))
683  {
691  EDMA_DRV_InstallCallback(&dspiState->edmaTxChannel,
692  DSPI_DRV_EdmaTxCallback,(void *)instance);
696  EDMA_HAL_SetDmaRequestCmd(dmaBaseAddr, (edma_channel_indicator_t)dmaChannel, true);
697 
699  DSPI_HAL_SetTxFifoFillDmaIntMode(base, kDspiGenerateDmaReq, true);
700  }
707  else
708  {
710  EDMA_HAL_SetDmaRequestCmd(dmaBaseAddr, (edma_channel_indicator_t)dmaChannel, false);
711 
713  DSPI_HAL_SetTxFifoFillDmaIntMode(base, kDspiGenerateDmaReq, false);
714 
716  EDMA_HAL_TriggerChannelStart(dmaBaseAddr, (edma_channel_indicator_t)dmaChannel);
717  }
718  }
719  if(dspiState->remainingReceiveByteCount > 0)
720  {
722  DSPI_HAL_SetRxFifoDrainDmaIntMode(base, kDspiGenerateDmaReq, true);
723  }
724 
726  dspiState->isTransferInProgress = true;
727  dspiState->status = kStatus_DSPI_Busy;
728 }

760 /*FUNCTION**********************************************************************
761  *
762  * Function Name : DSPI_DRV_EdmaCompleteTransfer
763  * Description : Finish the transfer
764  * Called when transfer is finished
765  *
766  *END*/
767 static void DSPI_DRV_EdmaCompleteTransfer(uint32_t instance)
768 {
769  dspi_edma_slave_state_t * dspiState = (dspi_edma_slave_state_t *)g_dspiStatePtr[instance];
770  SPI_Type *base = g_dspiBase[instance];
771  DMA_Type * edmaBaseAddr = VIRTUAL_CHN_TO_EDMA_MODULE_REGBASE(dspiState->edmaRxChannel.channel);
772  uint32_t edmaChannel = VIRTUAL_CHN_TO_EDMA_CHN(dspiState->edmaRxChannel.channel);
773  uint32_t readData;
774 
776  EDMA_HAL_HTCDSetIntCmd(edmaBaseAddr, edmaChannel, false);
777 
779  EDMA_HAL_SetDmaRequestCmd(edmaBaseAddr, (edma_channel_indicator_t)edmaChannel, false);
780 
781  edmaBaseAddr = VIRTUAL_CHN_TO_EDMA_MODULE_REGBASE(dspiState->edmaTxChannel.channel);
782  edmaChannel = VIRTUAL_CHN_TO_EDMA_CHN(dspiState->edmaTxChannel.channel);
783 
785  EDMA_HAL_HTCDSetIntCmd(edmaBaseAddr, edmaChannel, false);
786 
788  EDMA_HAL_SetDmaRequestCmd(edmaBaseAddr, (edma_channel_indicator_t)edmaChannel, false);
789 
791  DSPI_HAL_StopTransfer(base);
792 
794  DSPI_HAL_SetTxFifoFillDmaIntMode(base, kDspiGenerateDmaReq, false);
796  DSPI_HAL_SetRxFifoDrainDmaIntMode(base, kDspiGenerateDmaReq, false);
798  DSPI_HAL_SetIntMode(base, kDspiTxComplete, false);
799 
801  if((dspiState->extraReceiveByte > 0) && (dspiState->receiveBuffer))
802  {
804  readData = DSPI_HAL_ReadData(base);
805 
807  dspiState->receiveBuffer[dspiState->remainingReceiveByteCount] = (uint8_t)readData;
808  if ((dspiState->extraReceiveByte > 0) &&(--dspiState->extraReceiveByte > 0))
809  {
811  dspiState->receiveBuffer[dspiState->remainingReceiveByteCount + 1] = (uint8_t)(readData >> 8);
812  }
813  }
814 
816  dspiState->status = kStatus_DSPI_Success;
817  dspiState->isTransferInProgress = false;
818  dspiState->sendBuffer = NULL;
819  dspiState->receiveBuffer = NULL;
820  dspiState->remainingSendByteCount = 0;
821  dspiState->remainingReceiveByteCount = 0;
822  dspiState->extraReceiveByte = 0;
823 }

825 /*FUNCTION**********************************************************************
826  *
827  * Function Name : DSPI_DRV_EdmaSlaveGetTransferStatus
