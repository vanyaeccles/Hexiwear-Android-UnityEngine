79 /*FUNCTION**********************************************************************
80  *
81  * Function Name : I2C_DRV_MasterInit
82  * Description : initializes the I2C master mode driver.
83  * This function will initialize the I2C master mode driver, enable I2C clock,
84  * and enable I2C interrupt.
85  *
86  *END*/
87 i2c_status_t I2C_DRV_MasterInit(uint32_t instance, i2c_master_state_t * master)
88 {
89  assert(master);
90  assert(instance < I2C_INSTANCE_COUNT);
91 
92  I2C_Type * base = g_i2cBase[instance];
93 
95  if (g_i2cStatePtr[instance])
96  {
97  return kStatus_I2C_Initialized;
98  }
99 
101  memset(master, 0, sizeof(i2c_master_state_t));
102 
104  OSA_SemaCreate(&master->irqSync, 0);
105 
107  CLOCK_SYS_EnableI2cClock(instance);
108 
110  I2C_HAL_Init(base);
111 
113  g_i2cStatePtr[instance] = master;
114 
116  INT_SYS_EnableIRQ(g_i2cIrqId[instance]);
117 
119  master->i2cIdle = true;
120 
122  I2C_HAL_Enable(base);
123 
124  return kStatus_I2C_Success;
125 }

127 /*FUNCTION**********************************************************************
128  *
129  * Function Name : I2C_DRV_MasterDeinit
130  * Description : Deinit the I2C master mode driver.
131  * This function will deinit the I2C master mode driver, disable I2C clock,
132  * and disable I2C interrupt.

134  *END*/
135 i2c_status_t I2C_DRV_MasterDeinit(uint32_t instance)
136 {
137  assert(instance < I2C_INSTANCE_COUNT);
138 
140  if ((!g_i2cStatePtr[instance]) || (!CLOCK_SYS_GetI2cGateCmd(instance)))
141  {
142  return kStatus_I2C_Fail;
143  }
144 
145  I2C_Type * base = g_i2cBase[instance];
146  i2c_master_state_t * master = (i2c_master_state_t *)g_i2cStatePtr[instance];
147 
149  I2C_HAL_Disable(base);
150 
152  CLOCK_SYS_DisableI2cClock(instance);
153 
155  INT_SYS_DisableIRQ(g_i2cIrqId[instance]);
156 
158  OSA_SemaDestroy(&master->irqSync);
159 
161  g_i2cStatePtr[instance] = NULL;
162 
163  return kStatus_I2C_Success;

348 /*FUNCTION**********************************************************************
349  *
350  * Function Name : I2C master IRQ handler.
351  * Description : This handler uses the buffers stored in the
352  * i2c_master_state_t structs to transfer data.
353  * This is not a public API as it is called whenever an interrupt occurs.
354  *
355  *END*/
356 void I2C_DRV_MasterIRQHandler(uint32_t instance)
357 {
358  assert(instance < I2C_INSTANCE_COUNT);
359 
360  I2C_Type * base = g_i2cBase[instance];
361 
363  I2C_HAL_ClearInt(base);
364 
366  i2c_master_state_t * master = (i2c_master_state_t *)g_i2cStatePtr[instance];
367 
369  i2c_direction_t direction = I2C_HAL_GetDirMode(base);
370 
372  if ((!I2C_HAL_GetStatusFlag(base, kI2CBusBusy)) ||
373  (!I2C_HAL_IsMaster(base)))
374  {
375  return;
376  }
377 
379  if (direction == kI2CSend)
380  {
382  if (I2C_HAL_GetStatusFlag(base, kI2CReceivedNak))
383  {
385  master->status = kStatus_I2C_ReceivedNak;
386 
388  I2C_DRV_CompleteTransfer(instance);
389  }
390  else
391  {
395  if (--master->txSize > 0)
396  {
398  I2C_HAL_WriteByte(base, *(++master->txBuff));
399  }
400  else
401  {
403  I2C_DRV_CompleteTransfer(instance);
404  }
405  }
406  }
407  else
408  {
409  switch (--master->rxSize)
410  {
411  case 0x0U:
413  I2C_DRV_CompleteTransfer(instance);
414  break;
415  case 0x1U:
417  I2C_HAL_SendNak(base);
418  break;
419  default :
420  I2C_HAL_SendAck(base);
421  break;
422  }
423 
425  *(master->rxBuff++) = I2C_HAL_ReadByte(base);
426  }
427 }

429 /*FUNCTION**********************************************************************
430  *
431  * Function Name : I2C_DRV_MasterWait
432  * Description : Wait transfer to finish.
433  * This function is a static function which will be called by other data
434  * transaction APIs.
435  *
436  *END*/
437 static i2c_status_t I2C_DRV_MasterWait(uint32_t instance, uint32_t timeout_ms)
438 {
439  assert(instance < I2C_INSTANCE_COUNT);
440 
441  i2c_master_state_t * master = (i2c_master_state_t *)g_i2cStatePtr[instance];
442  osa_status_t syncStatus;
443 
444  do
445  {
446  syncStatus = OSA_SemaWait(&master->irqSync, timeout_ms);
447  }while(syncStatus == kStatus_OSA_Idle);
448 
449  if (syncStatus != kStatus_OSA_Success)
450  {
451  master->status = kStatus_I2C_Timeout;
452  }
453 
454  return master->status;
455 }

457 /*FUNCTION**********************************************************************
458  *
459  * Function Name : I2C_DRV_CompleteTransfer
460  * Description : Send STOP and disable interrupt when error happens or finish
461  * I2C transfers.
462  * This function is a static function which will be called by other data
463  * transaction APIs.
464  *
465  *END*/
466 static void I2C_DRV_CompleteTransfer(uint32_t instance)
467 {
468  assert(instance < I2C_INSTANCE_COUNT);
469 
470  I2C_Type * base = g_i2cBase[instance];
471  i2c_master_state_t * master = (i2c_master_state_t *)g_i2cStatePtr[instance];
472 
473  if ((!master->isRequesting)
474  || (master->status == kStatus_I2C_ReceivedNak)
475  || (master->status == kStatus_I2C_Timeout))
476  {
478  I2C_HAL_SetIntCmd(base, false);
479 
481  I2C_HAL_SendStop(base);
482 
484  master->i2cIdle = true;
485  }
486 
487  if (master->isBlocking)
488  {
489  OSA_SemaPost(&master->irqSync);
490  }
491 }

493 /*FUNCTION**********************************************************************
494  *
495  * Function Name : I2C_DRV_SendAddress
496  * Description : Prepare and send out address buffer with interrupt.
497  * This function is a static function which will be called by other data
498  * transaction APIs.
499  *
500  *END*/
501 static i2c_status_t I2C_DRV_SendAddress(uint32_t instance,
502  const i2c_device_t * device,
503  const uint8_t * cmdBuff,
504  uint32_t cmdSize,
505  i2c_direction_t direction,
506  uint32_t timeout_ms)
507 {
508  assert(instance < I2C_INSTANCE_COUNT);
509 
510  I2C_Type * base = g_i2cBase[instance];
512  i2c_master_state_t * master = (i2c_master_state_t *)g_i2cStatePtr[instance];
513 
514  uint8_t addrByte1, addrByte2, directionBit;
515  bool is10bitAddr;
516  uint8_t addrBuff[2] = {0};
517  uint8_t addrSize = 0;
518  bool isMainXferBlocking = master->isBlocking;
519 
521  master->isRequesting = true;
522  master->isBlocking = true;
523 
524  /*--------------- Prepare Address Buffer ------------------*/
527  directionBit = (direction == kI2CReceive) ? 0x1U : 0x0U;
528 
530  is10bitAddr = ((device->address >> 10U) == 0x1EU) ? true : false;
531 
533  if (is10bitAddr)
534  {
535  addrByte1 = (uint8_t)(device->address >> 8U);
536  addrByte2 = (uint8_t)device->address;
537  }
538  else
539  {
540  addrByte1 = (uint8_t)device->address;
541  }
542 
546  addrByte1 = addrByte1 << 1U;
547 
549  addrByte1 |= (uint8_t)((is10bitAddr || cmdBuff) ? 0U : directionBit);
550 
552  addrBuff[addrSize++] = addrByte1;
553 
554  if (is10bitAddr)
555  {
557  addrBuff[addrSize++] = addrByte2;
558  }
559 
560  /*--------------- Send Address Buffer ------------------*/
561  master->txBuff = addrBuff;
562  master->txSize = addrSize;
563 
565  I2C_HAL_WriteByte(base, addrBuff[0]);
566 
568  I2C_DRV_MasterWait(instance, timeout_ms);
569 
570  /*--------------------- Send CMD -----------------------*/
571  if ((master->status == kStatus_I2C_Success) && cmdBuff)
572  {
573  master->txBuff = cmdBuff;
574  master->txSize = cmdSize;
575 
577  I2C_HAL_WriteByte(base, *cmdBuff);
578 
580  I2C_DRV_MasterWait(instance, timeout_ms);
581  }
582 
583  /*--------------- Send Address Again ------------------*/
586  if ((master->status == kStatus_I2C_Success) && (direction == kI2CReceive)
587  && (is10bitAddr || cmdBuff))
588  {
590  master->txSize = 1U;
591  master->txBuff = NULL;
592 
594  I2C_HAL_SendStart(base);
595 
597  I2C_HAL_WriteByte(base, (uint8_t)(addrByte1 | 1U));
598 
600  I2C_DRV_MasterWait(instance, timeout_ms);
601  }
602 
603  master->isRequesting = false;
604  master->isBlocking = isMainXferBlocking ;
605 
606  return master->status;
607 }

609 /*FUNCTION**********************************************************************
610  *
611  * Function Name : I2C_DRV_MasterSend
612  * Description : Private function to handle blocking/non-blocking send.
613  * This function is a static function which will be called by other data
614  * transaction APIs.
615  *
616  *END*/
617 static i2c_status_t I2C_DRV_MasterSend(uint32_t instance,
618  const i2c_device_t * device,
619  const uint8_t * cmdBuff,
620  uint32_t cmdSize,
621  const uint8_t * txBuff,
622  uint32_t txSize,
623  uint32_t timeout_ms,
624  bool isBlocking)
625 {
626  assert(instance < I2C_INSTANCE_COUNT);
627  assert(txBuff);
628 
629  I2C_Type * base = g_i2cBase[instance];
630  i2c_master_state_t * master = (i2c_master_state_t *)g_i2cStatePtr[instance];
631 
633  if (!master->i2cIdle)
634  {
635  return master->status = kStatus_I2C_Busy;
636  }
637 
639  if (!isBlocking)
640  {
641  timeout_ms = I2C_TIMEOUT_MS;
642  }
643 
644  master->txBuff = NULL;
645  master->txSize = 0;
646  master->rxBuff = NULL;
647  master->rxBuff = 0;
648  master->status = kStatus_I2C_Success;
649  master->i2cIdle = false;
650  master->isBlocking = isBlocking;
651 
652  I2C_DRV_MasterSetBaudRate(instance, device);
653 
655  I2C_HAL_SetDirMode(base, kI2CSend);
656 
658  I2C_HAL_ClearInt(base);
659  I2C_HAL_SetIntCmd(base, true);
660 
662  I2C_HAL_SendStart(base);
663 
665  I2C_DRV_SendAddress(instance, device, cmdBuff, cmdSize, kI2CSend, timeout_ms);
666 
668  if (master->status == kStatus_I2C_Success)
669  {
671  master->txBuff = txBuff;
672  master->txSize = txSize;
673 
675  I2C_HAL_WriteByte(base, master->txBuff[0]);
676 
677  if (isBlocking)
678  {
680  I2C_DRV_MasterWait(instance, timeout_ms);
681  }
682  }
683  else if (master->status == kStatus_I2C_Timeout)
684  {
686  I2C_HAL_SetIntCmd(base, false);
687 
688  if (I2C_HAL_GetStatusFlag(base, kI2CBusBusy))
689  {
691  I2C_HAL_SendStop(base);
692  }
693 
695  master->i2cIdle = true;
696  }
697 
698  return master->status;
699 }

787  I2C_HAL_SetIntCmd(base, false);
