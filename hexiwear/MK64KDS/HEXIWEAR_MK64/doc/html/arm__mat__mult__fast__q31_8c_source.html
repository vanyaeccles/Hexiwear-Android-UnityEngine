86  arm_matrix_instance_q31 * pDst)
87 {
88  q31_t *pIn1 = pSrcA->pData;
89  q31_t *pIn2 = pSrcB->pData;
90  q31_t *pInA = pSrcA->pData;
91 // q31_t *pSrcB = pSrcB->pData; /** input data matrix pointer B */
92  q31_t *pOut = pDst->pData;
93  q31_t *px;
94  q31_t sum;
95  uint16_t numRowsA = pSrcA->numRows;
96  uint16_t numColsB = pSrcB->numCols;
97  uint16_t numColsA = pSrcA->numCols;
98  uint16_t col, i = 0u, j, row = numRowsA, colCnt;
99  arm_status status;
100  q31_t inA1, inA2, inA3, inA4, inB1, inB2, inB3, inB4;
101 
102 #ifdef ARM_MATH_MATRIX_CHECK
103 
104 
106  if((pSrcA->numCols != pSrcB->numRows) ||
107  (pSrcA->numRows != pDst->numRows) || (pSrcB->numCols != pDst->numCols))
108  {
110  status = ARM_MATH_SIZE_MISMATCH;
111  }
112  else
113 #endif /* #ifdef ARM_MATH_MATRIX_CHECK */
114 
115  {
118  do
119  {
121  px = pOut + i;
122 
124  col = numColsB;
125 
128  pIn2 = pSrcB->pData;
129 
130  j = 0u;
131 
133  do
134  {
136  sum = 0;
137 
139  pIn1 = pInA;
140 
142  colCnt = numColsA >> 2;
143 
144 
146  while(colCnt > 0u)
147  {
150  inB1 = *pIn2;
151  pIn2 += numColsB;
152 
153  inA1 = pIn1[0];
154  inA2 = pIn1[1];
155 
156  inB2 = *pIn2;
157  pIn2 += numColsB;
158 
159  inB3 = *pIn2;
160  pIn2 += numColsB;
161 
162  sum = (q31_t) ((((q63_t) sum << 32) + ((q63_t) inA1 * inB1)) >> 32);
163  sum = (q31_t) ((((q63_t) sum << 32) + ((q63_t) inA2 * inB2)) >> 32);
164 
165  inA3 = pIn1[2];
166  inA4 = pIn1[3];
167 
168  inB4 = *pIn2;
169  pIn2 += numColsB;
170 
171  sum = (q31_t) ((((q63_t) sum << 32) + ((q63_t) inA3 * inB3)) >> 32);
172  sum = (q31_t) ((((q63_t) sum << 32) + ((q63_t) inA4 * inB4)) >> 32);
173 
174  pIn1 += 4u;
175 
177  colCnt--;
178  }
179 
182  colCnt = numColsA % 0x4u;
183 
184  while(colCnt > 0u)
185  {
188  sum = (q31_t) ((((q63_t) sum << 32) +
189  ((q63_t) * pIn1++ * (*pIn2))) >> 32);
190  pIn2 += numColsB;
191 
193  colCnt--;
194  }
195 
197  *px++ = sum << 1;
198 
200  j++;
201  pIn2 = pSrcB->pData + j;
202 
204  col--;
205 
206  } while(col > 0u);
207 
209  i = i + numColsB;
210  pInA = pInA + numColsA;
211 
213  row--;
214 
215  } while(row > 0u);
216 
218  status = ARM_MATH_SUCCESS;
219  }
221  return (status);
222 }
