50 /*FUNCTION**********************************************************************
51  *
52  * Function Name : LPTMR_DRV_Init
53  * Description : initializes the LPTMR driver.
54  * This function will initialize the LPTMR driver according to user configure
55  * strcuture.
56  *
57  *END*/
58 lptmr_status_t LPTMR_DRV_Init(uint32_t instance, lptmr_state_t *userStatePtr, const lptmr_user_config_t* userConfigPtr)
59 {
60  assert(instance < LPTMR_INSTANCE_COUNT);
61 
62  LPTMR_Type * base = g_lptmrBase[instance];
63  lptmr_prescaler_user_config_t prescalerUserConfig;
64  lptmr_working_mode_user_config_t workingModeUserConfig;
65 
66  if ((!userConfigPtr) || (!userStatePtr))
67  {
68  return kStatus_LPTMR_NullArgument;
69  }
70 
72  if ((kLptmrTimerModePulseCounter == userConfigPtr->timerMode) &&
73  (true == userConfigPtr->prescalerEnable) &&
74  (kLptmrPrescalerDivide2 == userConfigPtr->prescalerValue))
75  {
76  return kStatus_LPTMR_InvalidPrescalerValue;
77  }
78 
80  CLOCK_SYS_EnableLptmrClock(instance);
81 
83  LPTMR_HAL_Disable(base);
84 
86  prescalerUserConfig.prescalerClockSelect = (lptmr_prescaler_clock_select_t)userConfigPtr->prescalerClockSource;
87  prescalerUserConfig.prescalerBypass = (uint8_t)(userConfigPtr->prescalerEnable == false);
88  prescalerUserConfig.prescalerValue = userConfigPtr->prescalerValue;
89  LPTMR_HAL_SetPrescalerMode(base, prescalerUserConfig);
90 
92  workingModeUserConfig.timerModeSelect = userConfigPtr->timerMode;
93  workingModeUserConfig.freeRunningEnable = userConfigPtr->freeRunningEnable;
94  workingModeUserConfig.pinPolarity = userConfigPtr->pinPolarity;
95  workingModeUserConfig.pinSelect = userConfigPtr->pinSelect;
96  LPTMR_HAL_SetTimerWorkingMode(base,workingModeUserConfig);
97 
99  lptmr_state_ptrs[instance] = userStatePtr;
100 
101  userStatePtr->userCallbackFunc = NULL;
102 
104  if (userConfigPtr->isInterruptEnabled)
105  {
106  LPTMR_HAL_SetIntCmd(base,true);
107  INT_SYS_EnableIRQ(g_lptmrIrqId[instance]);
108  }
109  else
110  {
111  LPTMR_HAL_SetIntCmd(base,false);
112  INT_SYS_DisableIRQ(g_lptmrIrqId[instance]);
113  }
114 
116  if ( kLptmrTimerModeTimeCounter == userConfigPtr->timerMode)
117  {
118  userStatePtr->prescalerClockHz = CLOCK_SYS_GetLptmrFreq(instance,
119  userConfigPtr->prescalerClockSource);
120 
121  if (userConfigPtr->prescalerEnable)
122  {
123  userStatePtr->prescalerClockHz = (userStatePtr->prescalerClockHz >> ((uint32_t)(userConfigPtr->prescalerValue+1)));
124  }
125  }
126 
127  return kStatus_LPTMR_Success;
128 }

201 /*FUNCTION**********************************************************************
202  *
203  * Function Name : LPTMR_DRV_SetTimerPeriodUs
204  * Description : Set LPTMR timer counter period with unit microsecond
205  * This function is used to set LPTMR timer counter period with unit microsecond
206  *
207  *END*/
208 lptmr_status_t LPTMR_DRV_SetTimerPeriodUs(uint32_t instance, uint32_t us)
209 {
210  assert(instance < LPTMR_INSTANCE_COUNT);
211  assert(us > 0);
212  assert(CLOCK_SYS_GetLptmrGateCmd(instance));
213 
214  LPTMR_Type * base = g_lptmrBase[instance];
215  uint32_t tick_count;
216 
217  if (lptmr_state_ptrs[instance]->prescalerClockHz < 1000000U)
218  {
219  if (us < (1000000U/lptmr_state_ptrs[instance]->prescalerClockHz))
220  {
221  return kStatus_LPTMR_TimerPeriodUsTooSmall;
222  }
223  else
224  {
225  tick_count = (us/(1000000U/lptmr_state_ptrs[instance]->prescalerClockHz));
226 
228  if ( tick_count > 0xFFFFU )
229  {
230  return kStatus_LPTMR_TimerPeriodUsTooLarge;
231  }
232  else
233  {
234  LPTMR_HAL_SetCompareValue(base,tick_count);
235  }
236  }
237  }
238  else
239  {
240  tick_count = (us*(lptmr_state_ptrs[instance]->prescalerClockHz/1000000U));
241 
243  if ( tick_count > 0xFFFFU )
244  {
245  return kStatus_LPTMR_TimerPeriodUsTooLarge;
246  }
247  else
248  {
249  LPTMR_HAL_SetCompareValue(base,tick_count);
250  }
251  }
252 
253  return kStatus_LPTMR_Success;
254 }

256 /*FUNCTION**********************************************************************
257  *
258  * Function Name : LPTMR_DRV_GetCurrentTimeUs
259  * Description : Get LPTMR current time with unit microsecond
260  * This function is used to get LPTMR current time with unit microsecond
261  *
262  *END*/
263 uint32_t LPTMR_DRV_GetCurrentTimeUs(uint32_t instance)
264 {
265  assert(instance < LPTMR_INSTANCE_COUNT);
266  assert(CLOCK_SYS_GetLptmrGateCmd(instance));
267 
268  LPTMR_Type * base = g_lptmrBase[instance];
269 
270  uint32_t us;
271 
272  if (lptmr_state_ptrs[instance]->prescalerClockHz < 1000000U)
273  {
274  us = LPTMR_HAL_GetCounterValue(base)*(1000000U/lptmr_state_ptrs[instance]->prescalerClockHz);
275  }
276  else
277  {
278  us = LPTMR_HAL_GetCounterValue(base)/(lptmr_state_ptrs[instance]->prescalerClockHz/1000000U);
279  }
280 
281  return us;
282 }

350 /*FUNCTION*********************************************************************
351  *
352  * Function Name : LPTMR_DRV_IRQHandler
353  * Description : The driver-defined ISR in LPTMR module.
354  * It includes the process for interrupt mode defined by driver. Currently, it
355  * will be called inside the system-defined ISR.
356  *
357  *END*/
358 void LPTMR_DRV_IRQHandler(uint32_t instance)
359 {
360  assert(instance < LPTMR_INSTANCE_COUNT);
361  assert(CLOCK_SYS_GetLptmrGateCmd(instance));
362 
363  LPTMR_Type * base = g_lptmrBase[instance];
364 
366  LPTMR_HAL_ClearIntFlag(base);
367 
368  if (lptmr_state_ptrs[instance])
369  {
370  if (lptmr_state_ptrs[instance]->userCallbackFunc)
371  {
373  (*(lptmr_state_ptrs[instance]->userCallbackFunc))();
374  }
375  }
376 }
