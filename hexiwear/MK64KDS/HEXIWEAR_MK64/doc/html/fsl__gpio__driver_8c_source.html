41 /*FUNCTION**********************************************************************
42  *
43  * Function Name : GPIO_DRV_Init
44  * Description : Initialize all GPIO pins used by board.
45  * To initialize the GPIO driver, two arrays similar with
46  * gpio_input_pin_user_config_t inputPin[] and
47  * gpio_output_pin_user_config_t outputPin[] should be defined in user's file.
48  * Then simply call GPIO_DRV_Init() and pass into these two arrays. If input
49  * or output pins is not needed, pass in a NULL.
50  *
51  *END*/
52 void GPIO_DRV_Init(const gpio_input_pin_user_config_t * inputPins,
53  const gpio_output_pin_user_config_t * outputPins)
54 {
55  if (inputPins)
56  {
58  while (inputPins->pinName != GPIO_PINS_OUT_OF_RANGE)
59  {
60  GPIO_DRV_InputPinInit(inputPins++);
61  }
62  }
63 
64  if (outputPins)
65  {
67  while (outputPins->pinName != GPIO_PINS_OUT_OF_RANGE)
68  {
69  GPIO_DRV_OutputPinInit(outputPins++);
70  }
71  }
72 }

74 /*FUNCTION**********************************************************************
75  *
76  * Function Name : GPIO_DRV_InputPinInit
77  * Description : Initialize one GPIO input pin used by board.
78  *
79  *END*/
80 void GPIO_DRV_InputPinInit(const gpio_input_pin_user_config_t *inputPin)
81 {
83  uint32_t port = GPIO_EXTRACT_PORT(inputPin->pinName);
84  uint32_t pin = GPIO_EXTRACT_PIN(inputPin->pinName);
85  GPIO_Type * gpioBase = g_gpioBase[port];
86  PORT_Type * portBase = g_portBase[port];
87 
89  CLOCK_SYS_EnablePortClock(port);
90 
92  PORT_HAL_SetMuxMode(portBase, pin, kPortMuxAsGpio);
93 
95  GPIO_HAL_SetPinDir(gpioBase, pin, kGpioDigitalInput);
96 
98  #if FSL_FEATURE_PORT_HAS_PULL_ENABLE
99  PORT_HAL_SetPullCmd(portBase, pin, inputPin->config.isPullEnable);
100  #endif
101  #if FSL_FEATURE_PORT_HAS_PULL_SELECTION
102  PORT_HAL_SetPullMode(portBase, pin, inputPin->config.pullSelect);
103  #endif
104  #if FSL_FEATURE_PORT_HAS_PASSIVE_FILTER
105  PORT_HAL_SetPassiveFilterCmd(portBase, pin,
106  inputPin->config.isPassiveFilterEnabled);
107  #endif
108  #if FSL_FEATURE_PORT_HAS_DIGITAL_FILTER
109  PORT_HAL_SetDigitalFilterCmd(portBase, pin,
110  inputPin->config.isDigitalFilterEnabled);
111  #endif
112  #if FSL_FEATURE_GPIO_HAS_INTERRUPT_VECTOR
113  PORT_HAL_SetPinIntMode(portBase, pin, inputPin->config.interrupt);
114 
116  if ((inputPin->config.interrupt) && (g_portIrqId[port]))
117  {
119  INT_SYS_EnableIRQ(g_portIrqId[port]);
120  }
121  #endif
122 }

130 void GPIO_DRV_OutputPinInit(const gpio_output_pin_user_config_t *outputPin)
131 {

134  uint32_t pin = GPIO_EXTRACT_PIN(outputPin->pinName);
135  GPIO_Type * gpioBase = g_gpioBase[port];

137 
139  CLOCK_SYS_EnablePortClock(port);
140 
142  PORT_HAL_SetMuxMode(portBase, pin, kPortMuxAsGpio);
143 
145  GPIO_HAL_SetPinDir(gpioBase, pin, kGpioDigitalOutput);
146 
148  GPIO_HAL_WritePinOutput(gpioBase, pin, outputPin->config.outputLogic);
149  #if FSL_FEATURE_PORT_HAS_SLEW_RATE
150  PORT_HAL_SetSlewRateMode(portBase, pin, outputPin->config.slewRate);
151  #endif
152  #if FSL_FEATURE_PORT_HAS_DRIVE_STRENGTH
153  PORT_HAL_SetDriveStrengthMode(portBase, pin, outputPin->config.driveStrength);
154  #endif
155  #if FSL_FEATURE_PORT_HAS_OPEN_DRAIN
156  PORT_HAL_SetOpenDrainCmd(portBase, pin, outputPin->config.isOpenDrainEnabled);
157  #endif
158 }
