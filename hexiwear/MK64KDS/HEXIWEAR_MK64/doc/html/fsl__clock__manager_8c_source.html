81 /*FUNCTION**********************************************************************
82  *
83  * Function Name : CLOCK_SYS_Init
84  * Description : Install pre-defined clock configurations.
85  * This function installs the pre-defined clock configuration table to the
86  * clock manager.
87  *
88  *END*/
89 clock_manager_error_code_t CLOCK_SYS_Init(clock_manager_user_config_t const **clockConfigsPtr,
90  uint8_t configsNumber,
91  clock_manager_callback_user_config_t **callbacksPtr,
92  uint8_t callbacksNumber)
93 {
94  assert(NULL != clockConfigsPtr);
95  assert(NULL != callbacksPtr);
96 
97  CLOCK_SYS_LOCK_INIT();
98 
99  g_clockState.configTable = clockConfigsPtr;
100  g_clockState.clockConfigNum = configsNumber;
101  g_clockState.callbackConfig = callbacksPtr;
102  g_clockState.callbackNum = callbacksNumber;
103 
104  /*
105  * errorCallbackIndex is the index of the callback which returns error
106  * during clock mode switch. If all callbacks return success, then the
107  * errorCallbackIndex is callbacksnumber.
108  */
109  g_clockState.errorCallbackIndex = callbacksNumber;
110 
111  return kClockManagerSuccess;
112 }

114 /*FUNCTION**********************************************************************
115  *
116  * Function Name : CLOCK_SYS_UpdateConfiguration
117  * Description : Send notification and change system clock configuration.
118  * This function sends the notification to all callback functions, if all
119  * callbacks return OK or forceful policy is used, this function will change
120  * system clock configuration.
121  *
122  *END*/
123 clock_manager_error_code_t CLOCK_SYS_UpdateConfiguration(uint8_t targetConfigIndex,
124  clock_manager_policy_t policy)
125 {
126  uint8_t callbackIdx;
127  clock_manager_error_code_t ret = kClockManagerSuccess;
128 
129  clock_manager_callback_user_config_t* callbackConfig;
130 
131  clock_notify_struct_t notifyStruct;
132  notifyStruct.targetClockConfigIndex = targetConfigIndex;
133  notifyStruct.policy = policy;
134 
136  if (targetConfigIndex >= g_clockState.clockConfigNum)
137  {
138  return kClockManagerErrorOutOfRange;
139  }
140 
141  OSA_EnterCritical(kCriticalLockSched);
143  g_clockState.errorCallbackIndex = g_clockState.callbackNum;
144 
146  notifyStruct.notifyType = kClockManagerNotifyBefore;
147 
149  for (callbackIdx=0; callbackIdx<g_clockState.callbackNum; callbackIdx++)
150  {
151  callbackConfig = g_clockState.callbackConfig[callbackIdx];
152  if ((NULL != callbackConfig) &&
153  ((uint8_t)callbackConfig->callbackType & (uint8_t)kClockManagerNotifyBefore))
154  {
155  if (kClockManagerSuccess !=
156  (*callbackConfig->callback)(&notifyStruct,
157  callbackConfig->callbackData))
158  {
159  g_clockState.errorCallbackIndex = callbackIdx;
161  ret = kClockManagerErrorNotificationBefore;
162 
163  if (kClockManagerPolicyAgreement == policy)
164  {
165  break;
166  }
167  }
168  }
169  }
170 
172  if ((kClockManagerSuccess == ret) ||
173  (policy == kClockManagerPolicyForcible))
174  {
176  OSA_EnterCritical(kCriticalDisableInt);
177  CLOCK_SYS_SetConfiguration(g_clockState.configTable[targetConfigIndex]);
178 
179  g_clockState.curConfigIndex = targetConfigIndex;
180  OSA_ExitCritical(kCriticalDisableInt);
181 
182  notifyStruct.notifyType = kClockManagerNotifyAfter;
183 
184  for (callbackIdx=0; callbackIdx<g_clockState.callbackNum; callbackIdx++)
185  {
186  callbackConfig = g_clockState.callbackConfig[callbackIdx];
187  if ((NULL != callbackConfig) &&
188  ((uint8_t)callbackConfig->callbackType & (uint8_t)kClockManagerNotifyAfter))
189  {
190  if (kClockManagerSuccess !=
191  (*callbackConfig->callback)(&notifyStruct,
192  callbackConfig->callbackData))
193  {
194  g_clockState.errorCallbackIndex = callbackIdx;
196  ret = kClockManagerErrorNotificationAfter;
197 
198  if (kClockManagerPolicyAgreement == policy)
199  {
200  break;
201  }
202  }
203  }
204  }
205  }
206  else
207  {
208  notifyStruct.notifyType = kClockManagerNotifyRecover;
209  while (callbackIdx--)
210  {
211  callbackConfig = g_clockState.callbackConfig[callbackIdx];
212  if (NULL != callbackConfig)
213  {
214  (*callbackConfig->callback)(&notifyStruct,
215  callbackConfig->callbackData);
216  }
217  }
218  }
219 
220  OSA_ExitCritical(kCriticalLockSched);
221 
222  return ret;
223 }

266 mcglite_mode_error_t CLOCK_SYS_SetMcgliteMode(mcglite_config_t const *targetConfig)
267 {
268  uint32_t outFreq = 0U;
269  mcglite_mode_error_t ret = kMcgliteModeErrNone;
270 
271  assert(targetConfig->mcglite_mode < kMcgliteModeStop);
272 
274  switch (targetConfig->mcglite_mode)
275  {
276  case kMcgliteModeLirc8M:
277  ret = CLOCK_HAL_SetLircMode(MCG,
278  kMcgliteLircSel8M,
279  targetConfig->fcrdiv,
280  &outFreq);
281  break;
282  case kMcgliteModeLirc2M:
283  ret = CLOCK_HAL_SetLircMode(MCG,
284  kMcgliteLircSel2M,
285  targetConfig->fcrdiv,
286  &outFreq);
287  break;
288  case kMcgliteModeExt:
289  ret = CLOCK_HAL_SetExtMode(MCG, &outFreq);
290  break;
291  default:
292  ret = CLOCK_HAL_SetHircMode(MCG, &outFreq);
293  break;
294  }
295 
297  if (kMcgliteModeErrNone == ret)
298  {
300  CLOCK_HAL_SetHircCmd(MCG, targetConfig->hircEnableInNotHircMode);
301 
303  CLOCK_HAL_SetLircCmd(MCG, targetConfig->irclkEnable);
304  CLOCK_HAL_SetLircStopCmd(MCG, targetConfig->irclkEnableInStop);
305 
307  CLOCK_HAL_SetLircSelMode(MCG, targetConfig->ircs);
308 
310  CLOCK_HAL_SetLircDiv2(MCG, targetConfig->lircDiv2);
311  }
312 
313  return ret;
314 }

339 /*FUNCTION******************************************************************************
340  *
341  * Function name : CLOCK_SYS_SetMcgMode
342  * Description : This function sets MCG to some target mode defined by the configure
343  * structure, if can not switch to target mode directly, this function will choose
344  * the proper path. If external clock is used in the target mode, please make sure
345  * it is enabled, for example, if the external oscillator is used, please setup EREFS/HGO
346  * correctly and make sure OSCINIT is set.
347  * This function is used by clock dynamic setting, it only supports some specific
348  * mode transitions, including:
349  * 1. FEI ==> FEE/PEE
350  * 2. BLPI <==> FEE/PEE
351  * 3. Reconfigure PLL in PEE mode.
352  * 4. Reconfigure FLL in FEE mode.
353  *
354  *END*/
355 mcg_mode_error_t CLOCK_SYS_SetMcgMode(mcg_config_t const *targetConfig,
356  void (* fllStableDelay)(void))
357 {
358  uint32_t outClkFreq;
359 #if FSL_FEATURE_MCG_HAS_PLL
360 
361  mcg_modes_t curMode; // Current MCG mode.
362 #endif
363 
364 #if FSL_FEATURE_MCG_USE_OSCSEL
365  mcg_oscsel_select_t oscsel = targetConfig->oscsel;
366 #else
367  mcg_oscsel_select_t oscsel = kMcgOscselOsc;
368 #endif
369 
370 #if FSL_FEATURE_MCG_HAS_PLL
371 #if (FSL_FEATURE_MCG_HAS_PLL1 || FSL_FEATURE_MCG_HAS_EXTERNAL_PLL)
372  mcg_pll_clk_select_t pllcs = targetConfig->pllcs;
373 #else
374  mcg_pll_clk_select_t pllcs = kMcgPllClkSelPll0;
375 #endif
376 #endif
377 
378 #if FSL_FEATURE_MCG_HAS_PLL
379  curMode = CLOCK_HAL_GetMcgMode(MCG);
380 #endif
381 
382  if (kMcgModeBLPI == targetConfig->mcg_mode)
383  {
384 #if FSL_FEATURE_MCG_HAS_PLL
385  // If current mode is PEE mode, swith to FBE mode first.
386  if (kMcgModePEE == curMode)
387  {
388  CLOCK_SYS_SetMcgPeeToFbe();
389  }
390 #endif
391  // Change to FBI mode.
392  CLOCK_HAL_SetFbiMode(MCG,
393  targetConfig->drs,
394  targetConfig->ircs,
395  targetConfig->fcrdiv,
396  fllStableDelay,
397  &outClkFreq);
398 
399  // Enable low power mode to enter BLPI.
400  CLOCK_HAL_SetLowPowerModeCmd(MCG, true);
401  }
402  else if (kMcgModeFEE == targetConfig->mcg_mode)
403  {
404 #if FSL_FEATURE_MCG_HAS_PLL
405  // If current mode is PEE mode, swith to FBE mode first.
406  if (kMcgModePEE == curMode)
407  {
408  CLOCK_SYS_SetMcgPeeToFbe();
409  }
410 #endif
411  // Disalbe low power mode.
412  CLOCK_HAL_SetLowPowerModeCmd(MCG, false);
413  // Configure FLL in FBE mode then switch to FEE mode.
414  CLOCK_HAL_SetFbeMode(MCG,
415  oscsel,
416  targetConfig->frdiv,
417  targetConfig->dmx32,
418  targetConfig->drs,
419  fllStableDelay,
420  &outClkFreq);
421  // Change CLKS to enter FEE mode.
422  CLOCK_HAL_SetClkOutSrc(MCG, kMcgClkOutSrcOut);
423  while (CLOCK_HAL_GetClkOutStat(MCG) != kMcgClkOutStatFll) {}
424  }
425 #if FSL_FEATURE_MCG_HAS_PLL
426  else if (kMcgModePEE == targetConfig->mcg_mode)
427  {
428  /*
429  * If current mode is FEI/FEE/BLPI, then switch to FBE mode first.
430  * If current mode is PEE mode, which means need to reconfigure PLL,
431  * fist switch to PBE mode and configure PLL, then switch to PEE.
432  */
433 
434  if (kMcgModePEE != curMode)
435  {
436  // Disalbe low power mode.
437  CLOCK_HAL_SetLowPowerModeCmd(MCG, false);
438 
439  // Change to FBE mode.
440  CLOCK_HAL_SetFbeMode(MCG,
441  oscsel,
442  targetConfig->frdiv,
443  targetConfig->dmx32,
444  targetConfig->drs,
445  fllStableDelay,
446  &outClkFreq);
447  }
448 
449  // Change to PBE mode.
450  CLOCK_HAL_SetPbeMode(MCG,
451  oscsel,
452  pllcs,
453  targetConfig->prdiv0,
454  targetConfig->vdiv0,
455  &outClkFreq);
456 
457  // Set CLKS to enter PEE mode.
458  CLOCK_HAL_SetClkOutSrc(MCG, kMcgClkOutSrcOut);
459  while (CLOCK_HAL_GetClkOutStat(MCG) != kMcgClkOutStatPll) {}
460  }
461 #endif
462  else
463  {
464  return kMcgModeErrModeUnreachable;
465  }
466 
468  CLOCK_HAL_SetInternalRefClkEnableCmd(MCG, targetConfig->irclkEnable);
469  CLOCK_HAL_SetInternalRefClkEnableInStopCmd(MCG, targetConfig->irclkEnableInStop);
470 
472  if (targetConfig->irclkEnable)
473  {
474  if (kMcgIrcFast == targetConfig->ircs)
475  {
477  CLOCK_HAL_UpdateFastClkInternalRefDiv(MCG, targetConfig->fcrdiv);
478  }
479 
480  CLOCK_HAL_SetInternalRefClkMode(MCG, targetConfig->ircs);
481  while (targetConfig->ircs != CLOCK_HAL_GetInternalRefClkMode(MCG)) {}
482  }
483 
484 #if FSL_FEATURE_MCG_HAS_PLL
485 
486  if (targetConfig->pll0EnableInFllMode)
487  {
488  CLOCK_HAL_EnablePll0InFllMode(MCG,
489  targetConfig->prdiv0,
490  targetConfig->vdiv0,
491  targetConfig->pll0EnableInStop);
492  }
493  else
494  {
495  CLOCK_HAL_SetPll0EnableCmd(MCG, false);
496  }
497 #endif
498 
499  return kMcgModeErrNone;
500 }
501 
502 #endif

510 /*FUNCTION**********************************************************************
511  *
512  * Function Name : CLOCK_SYS_OscInit
513  * Description : Initialize OSC.
514  *
515  * This function initializes OSC according to configuration.
516  *
517  *END*/
518 clock_manager_error_code_t CLOCK_SYS_OscInit(uint32_t instance,
519  osc_user_config_t *config)
520 {
521  assert(instance < OSC_INSTANCE_COUNT);
522  uint32_t capacitorMask = 0U;
523 
524  if (kOscSrcOsc == config->erefs)
525  {
526  capacitorMask = (config->enableCapacitor2p ? kOscCapacitor2p : 0U) |
527  (config->enableCapacitor4p ? kOscCapacitor4p : 0U) |
528  (config->enableCapacitor8p ? kOscCapacitor8p : 0U) |
529  (config->enableCapacitor16p ? kOscCapacitor16p : 0U);
530  OSC_HAL_SetCapacitor(g_oscBase[instance], capacitorMask);
531  }
532 
533 #if FSL_FEATURE_MCGLITE_HAS_RANGE0
534  CLOCK_HAL_SetRange0Mode(MCG, config->range);
535 #endif
536 #if FSL_FEATURE_MCGLITE_HAS_HGO0
537  CLOCK_HAL_SetHighGainOsc0Mode(MCG, config->hgo);
538 #endif
539  CLOCK_HAL_SetExtRefSelMode0(MCG, config->erefs);
540 
541  CLOCK_SYS_SetOscerConfigration(instance, &(config->oscerConfig));
542 
544  if ((kOscSrcOsc == config->erefs) &&
545  (true == config->oscerConfig.enable))
546  {
547  while(!CLOCK_HAL_IsOscStable(MCG)){}
548  }
549 
550  g_xtal0ClkFreq = config->freq;
551 
552  return kClockManagerSuccess;
553 }
