113 void PORTC_IRQHandler(void)
114 {
115  PORT_HAL_ClearPortIntFlag(PORTC_BASE_PTR);
116 
117  watch_CheckBatteryStatus();
118 
119  RTC_UpdateAlarm();
120 
121  if ( true == FXOS_IsTapEnabled() )
122  {
123  switch ( power_GetCurrentMode() )
124  {
125  case POWER_CURRENT_MODE_NORMAL:
126  {
127  // regular work mode, do nothing
128  break;
129  }
130 
131  // if we were in shallow sleep mode, just turn the OLED ON
132  case POWER_CURRENT_MODE_SLEEP_SHALLOW:
133  {
134  HostInterface_SetWakingTouch();
135  power_SetCurrentMode( POWER_CURRENT_MODE_NORMAL );
136  break;
137  }
138 
139  case POWER_CURRENT_MODE_SLEEP_TOTAL:
140  {
141  HostInterface_SetWakingTouch();
142  power_SetCurrentMode( POWER_CURRENT_MODE_NORMAL );
143 
144  if ( POWER_STATUS_SUCCESS != power_SetWakeSource( POWER_WAKE_SRC_USER ) )
145  {
146  asm("nop");
147  }
148  break;
149  }
150 
156  case POWER_CURRENT_MODE_SLEEP_SENSOR_TAG:
157  {
158  HostInterface_SetWakingTouch();
159  power_SetCurrentMode( POWER_CURRENT_MODE_NORMAL );
160 
161  if ( POWER_STATUS_SUCCESS == power_SetWakeSource( POWER_WAKE_SRC_USER ) )
162  {
163  }
164  else
165  {
166  asm("nop");
167  }
168  break;
169  }
170  }
171  }
