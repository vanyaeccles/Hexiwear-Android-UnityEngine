78 /*FUNCTION**********************************************************************
79  *
80  * Function Name : UART_HAL_SetBaudRate
81  * Description : Configure the UART baud rate.
82  * This function programs the UART baud rate to the desired value passed in by
83  * the user. The user must also pass in the module source clock so that the
84  * function can calculate the baud rate divisors to their appropriate values.
85  *
86  *END*/
87 uart_status_t UART_HAL_SetBaudRate(UART_Type * base, uint32_t sourceClockInHz,
88  uint32_t baudRate)
89 {
91  uint16_t sbr;
92 
94  sbr = sourceClockInHz / (baudRate * 16);
95 
97  if ( (sbr > 0x1FFF) || (sbr < 1) )
98  {
100  return kStatus_UART_BaudRateCalculationError;
101  }
102 
104  UART_BWR_BDH_SBR(base, (uint8_t)(sbr >> 8));
105  UART_WR_BDL(base, (uint8_t)sbr);
106 
107 #if FSL_FEATURE_UART_HAS_BAUD_RATE_FINE_ADJUST_SUPPORT
108 
111  uint16_t brfa = (32*sourceClockInHz/(baudRate*16)) - 32*sbr;
112 
114  UART_BWR_C4_BRFA(base, brfa);
115 #endif
116 
117  return kStatus_UART_Success;
118 }

120 /*FUNCTION**********************************************************************
121  *
122  * Function Name : UART_HAL_SetBaudRateDivisor
123  * Description : Set the UART baud rate modulo divisor value.
124  * This function allows the user to program the baud rate divisor directly in
125  * situations where the divisor value is known. In this case, the user may not
126  * want to call the UART_HAL_SetBaudRate() function as the divisor is already
127  * known to them.
128  *
129  *END*/
130 void UART_HAL_SetBaudRateDivisor(UART_Type * base, uint16_t baudRateDivisor)
131 {
133  assert( (baudRateDivisor < 0x1FFF) && (baudRateDivisor > 1) );
134 
136  UART_BWR_BDH_SBR(base, (uint8_t)(baudRateDivisor >> 8));
137  UART_WR_BDL(base, (uint8_t)baudRateDivisor);
138 }

247 /*FUNCTION**********************************************************************
248  *
249  * Function Name : UART_HAL_SendDataPolling
250  * Description : Send out multiple bytes of data using polling method.
251  * This function only supports 8-bit transaction.
252  *
253  *END*/
254 void UART_HAL_SendDataPolling(UART_Type * base,
255  const uint8_t *txBuff,
256  uint32_t txSize)
257 {
258  while (txSize--)
259  {
260  while (!UART_BRD_S1_TDRE(base))
261  {}
262 
263  UART_HAL_Putchar(base, *txBuff++);
264  }
265 }

267 /*FUNCTION**********************************************************************
268  *
269  * Function Name : UART_HAL_ReceiveDataPolling
270  * Description : Receive multiple bytes of data using polling method.
271  * This function only supports 8-bit transaction.
272  *
273  *END*/
274 uart_status_t UART_HAL_ReceiveDataPolling(UART_Type * base,
275  uint8_t *rxBuff,
276  uint32_t rxSize)
277 {
278  uart_status_t retVal = kStatus_UART_Success;
279  uint8_t dummy = 0;
280 
281  while (rxSize--)
282  {
283  while (!UART_BRD_S1_RDRF(base))
284  {}
285 
286  UART_HAL_Getchar(base, rxBuff++);
287 
289  if (UART_BRD_S1_OR(base))
290  {
291  dummy = UART_RD_S1(base);
292  dummy = UART_RD_D(base);
293  dummy++;
294  retVal = kStatus_UART_RxOverRun;
295  }
296  }
297 
298  return retVal;
299 }

304 /*FUNCTION**********************************************************************
305  *
306  * Function Name : UART_HAL_ConfigureInterrupts
307  * Description : Configure the UART module interrupts to enable/disable various
308  * interrupt sources.
309  *
310  *END*/
311 void UART_HAL_SetIntMode(UART_Type * base, uart_interrupt_t interrupt, bool enable)
312 {
313  uint8_t reg = (uint32_t)interrupt >> UART_SHIFT;
314  uint32_t temp = 1U << (uint8_t)interrupt;
315 
316  switch ( reg )
317  {
318  case 0 :
319  enable ? UART_SET_BDH(base, temp) : UART_CLR_BDH(base, temp);
320  break;
321  case 1 :
322  enable ? UART_SET_C2(base, temp) : UART_CLR_C2(base, temp);
323  break;
324  case 2 :
325  enable ? UART_SET_C3(base, temp) : UART_CLR_C3(base, temp);
326  break;
327 #if FSL_FEATURE_UART_HAS_FIFO
328  case 3 :
329  enable ? UART_SET_CFIFO(base, temp) : UART_CLR_CFIFO(base, temp);
330  break;
331 #endif
332  default :
333  break;
334  }
335 }

443  *
444  * Function Name : UART_HAL_SetISO7816IntMode
445  * Description : Configure the UART module ISO7816 feature specific interrupts
446  * to enable/disable various
447  * interrupt sources.
448  *
449  *END*/
450 void UART_HAL_SetISO7816IntMode(UART_Type * base, uart_iso7816_interrupt_t interrupt, bool enable)
451 {
452  switch (interrupt)
453  {
454  case kUartIntIso7816RxThreasholdExceeded:
455  enable ? UART_BWR_IE7816_RXTE(base, 1U) : UART_BWR_IE7816_RXTE(base, 0U);
456  break;
457  case kUartIntIso7816TxThresholdExceeded:
458  enable ? UART_BWR_IE7816_TXTE(base, 1U) : UART_BWR_IE7816_TXTE(base, 0U);
459  break;
460  case kUartIntIso7816GuardTimerViolated:
461  enable ? UART_BWR_IE7816_GTVE(base, 1U) : UART_BWR_IE7816_GTVE(base, 0U);
462  break;
463 #if FSL_FEATURE_UART_HAS_IMPROVED_SMART_CARD_SUPPORT
464  case kUartIntIso7816AtrDurationTimer:
465  enable ? UART_BWR_IE7816_ADTE(base, 1U) : UART_BWR_IE7816_ADTE(base, 0U);
466  break;
467 #endif
468  case kUartIntIso7816InitialCharDetected:
469  enable ? UART_BWR_IE7816_INITDE(base, 1U) : UART_BWR_IE7816_INITDE(base, 0U);
470  break;
471  case kUartIntIso7816BlockWaitTimer:
472  enable ? UART_BWR_IE7816_BWTE(base, 1U) : UART_BWR_IE7816_BWTE(base, 0U);
473  break;
474  case kUartIntIso7816CharWaitTimer:
475  enable ? UART_BWR_IE7816_CWTE(base, 1U) : UART_BWR_IE7816_CWTE(base, 0U);
476  break;
477  case kUartIntIso7816WaitTimer:
478  enable ? UART_BWR_IE7816_WTE(base, 1U) : UART_BWR_IE7816_WTE(base, 0U);
479  break;
480  case kUartIntIso7816All:
481  enable ? UART_WR_IE7816(base, 0xFFU) : UART_WR_IE7816(base, 0x00U);
482  break;
483  default: break;
484  }
485 }

563 /*FUNCTION**********************************************************************
564  *
565  * Function Name : UART_HAL_SetISO7816Etu
566  * Description : Sets the basic Elementaty Time Unit of UART instance in ISO7816
567  * mode.
568  *
569  *END*/
570 uart_status_t UART_HAL_SetISO7816Etu(UART_Type * base, uint32_t sourceClockInHz, uint32_t sCClock, uint16_t Fi, uint8_t Di)
571 {
573  uint32_t brfa, sbr;
574  uint32_t sourceClockInKHz = sourceClockInHz/1000;
575  uint32_t sCClockInKHz = sCClock/1000;
576 
578  sbr = (uint16_t)(sourceClockInKHz * Fi/(Di* sCClockInKHz * 16));
579 
581  if ( (sbr > 0x1FFF) || (sbr < 1) )
582  {
584  return kStatus_UART_BaudRateCalculationError;
585  }
586 
587  UART_BWR_BDH_SBR(base, (uint8_t)(sbr >> 8));
588  UART_WR_BDL(base, (uint8_t)sbr);
589 
590 #if FSL_FEATURE_UART_HAS_BAUD_RATE_FINE_ADJUST_SUPPORT
591 
594  brfa = (32*sourceClockInKHz*Fi)/(Di*sCClockInKHz*16);
595  brfa -= (uint16_t)(32*sbr);
596 
598  UART_BWR_C4_BRFA(base, brfa);
599 #endif
600 
601  return kStatus_UART_Success;
602 }
