73 /*FUNCTION**********************************************************************
74  *
75  * Function Name : DSPI_DRV_EdmaMasterInit
76  * Description : Initializes a DSPI instance for master mode operation to work with EDMA.
77  * This function uses a dma driven method for transferring data.
78  * This function initializes the run-time state structure to track the ongoing
79  * transfers, ungates the clock to the DSPI module, resets the DSPI module, initializes the module
80  * to user defined settings and default settings, configures the IRQ state structure, enables
81  * the module-level interrupt to the core, and enables the DSPI module.
82  * The CTAR parameter is special in that it allows the user to have different SPI devices
83  * connected to the same DSPI module instance in addition to different peripheral chip
84  * selects. Each CTAR contains the bus attributes associated with that particular SPI device.
85  * For most use cases where only one SPI device is connected per DSPI module
86  * instance, use CTAR0.
87  * This is an example to set up and call the DSPI_DRV_EdmaMasterInit function by passing
88  * in these parameters:
89  * dspi_edma_master_state_t dspiEdmaState; <- the user simply allocates memory for this struct
90  * uint32_t calculatedBaudRate;
91  * dspi_edma_master_user_config_t userConfig; <- the user fills out members for this struct
92  * userConfig.isChipSelectContinuous = false;
93  * userConfig.isSckContinuous = false;
94  * userConfig.pcsPolarity = kDspiPcs_ActiveLow;
95  * userConfig.whichCtar = kDspiCtar0;
96  * userConfig.whichPcs = kDspiPcs0;

98  *
99  *END*/

365 /*FUNCTION**********************************************************************
366  *
367  * Function Name : DSPI_DRV_EdmaMasterSetDelay
368  * Description : Configures the DSPI master mode bus timing delay options with EDMA support.
369  *
370  * This function allows the user to take advantage of the DSPI module's delay options in order to
371  * "fine tune" some of the signal timings to match the timing needs of a slower peripheral device.
372  * This is an optional function that can be called after the DSPI module has been initialized for
373  * master mode.
374  * The bus timing delays that can be adjusted are listed below:
375  *
376  * PCS to SCK Delay: Adjustable delay option between the assertion of the PCS signal to the
377  * first SCK edge.
378  *
379  * After SCK Delay: Adjustable delay option between the last edge of SCK to the de-assertion
380  * of the PCS signal.
381  *
382  * Delay after Transfer: Adjustable delay option between the de-assertion of the PCS signal for a
383  * frame to the assertion of the PCS signal for the next frame. Note this
384  * is not adjustable for continuous clock mode as this delay is fixed at
385  * one SCK period.
386  *
387  * Each of the above delay parameters use both a pre-scalar and scalar value to calculate the
388  * needed delay. This function takes in as a parameter the desired delay type and the
389  * delay value (in nanoseconds), calculates the values needed for the prescaler and scaler.
390  * Returning the actual calculated delay as an exact delay match may not be possible. In this
391  * case, the closest match is calculated without going below the desired delay value input.
392  * It is possible to input a very large delay value that exceeds the capability of the part, in
393  * which case the maximum supported delay is returned. In addition, the function returns
394  * an out-of-range status.
395  *
396  *END*/
397 dspi_status_t DSPI_DRV_EdmaMasterSetDelay(uint32_t instance, dspi_delay_type_t whichDelay,
398  uint32_t delayInNanoSec, uint32_t * calculatedDelay)
399 {
401  dspi_edma_master_state_t * dspiEdmaState = (dspi_edma_master_state_t *)g_dspiStatePtr[instance];
402  SPI_Type *base = g_dspiBase[instance];
403  dspi_status_t errorCode = kStatus_DSPI_Success;
404 
405  *calculatedDelay = DSPI_HAL_CalculateDelay(base, dspiEdmaState->whichCtar, whichDelay,
406  dspiEdmaState->dspiSourceClock, delayInNanoSec);
407 
409  if (*calculatedDelay < delayInNanoSec)
410  {
411  errorCode = kStatus_DSPI_OutOfRange;
412  }
413 
414  return errorCode;
415 }

417 /*FUNCTION**********************************************************************
418  *
419  * Function Name : DSPI_DRV_EdmaMasterConfigureBus
420  * Description : Configures the DSPI port physical parameters to access a device on the bus with
421  * EDMA support.
422  *
423  * The term "device" is used to indicate the SPI device for which the DSPI master is communicating.
424  * The user has two options to configure the device parameters: either pass in the
425  * pointer to the device configuration structure to the desired transfer function (see
426  * DSPI_DRV_EdmaMasterTransferBlocking or DSPI_DRV_EdmaMasterTransfer) or pass it in to the
427  * DSPI_DRV_EdmaMasterConfigureBus function. The user can pass in a device structure to the
428  * transfer function which contains the parameters for the bus (the transfer function then calls
429  * this function). However, the user has the option to call this function directly especially
430  * to get the calculated baud rate, at which point they may pass in NULL for the device
431  * structure in the transfer function (assuming they have called this configure bus function
432  * first). This is an example to set up the dspi_device_t structure to call
433  * the DSPI_DRV_EdmaMasterConfigureBus function by passing in these parameters:
434  * dspi_edma_device_t spiDevice;
435  * spiDevice.dataBusConfig.bitsPerFrame = 16;
436  * spiDevice.dataBusConfig.clkPhase = kDspiClockPhase_FirstEdge;
437  * spiDevice.dataBusConfig.clkPolarity = kDspiClockPolarity_ActiveHigh;
438  * spiDevice.dataBusConfig.direction = kDspiMsbFirst;
439  * spiDevice.bitsPerSec = 50000;
440  * DSPI_DRV_EdmaMasterConfigureBus(instance, &spiDevice, &calculatedBaudRate);
441  *
442  *END*/
443 dspi_status_t DSPI_DRV_EdmaMasterConfigureBus(uint32_t instance,
444  const dspi_edma_device_t * device,
445  uint32_t * calculatedBaudRate)
446 {
447  assert(device);
449  dspi_edma_master_state_t * dspiEdmaState = (dspi_edma_master_state_t *)g_dspiStatePtr[instance];
450  SPI_Type *base = g_dspiBase[instance];
451 
452  dspi_status_t errorCode = kStatus_DSPI_Success;
453 
455  *calculatedBaudRate = DSPI_HAL_SetBaudRate(base, dspiEdmaState->whichCtar,
456  device->bitsPerSec, dspiEdmaState->dspiSourceClock);
457 
458  errorCode = DSPI_HAL_SetDataFormat(base, dspiEdmaState->whichCtar, &device->dataBusConfig);
459 
461  if (device->dataBusConfig.bitsPerFrame > 16)
462  {
463  errorCode = kStatus_DSPI_OutOfRange;
464  }
465  else
466  {
467  dspiEdmaState->bitsPerFrame = device->dataBusConfig.bitsPerFrame;
468  }
469 
470  return errorCode;
471 }

473 /*FUNCTION**********************************************************************
474  *
475  * Function Name : DSPI_DRV_EdmaMasterTransferBlocking
476  * Description : Performs a blocking SPI master mode transfer with EDMA support.
477  *
478  * This function simultaneously sends and receives data on the SPI bus, as SPI is naturally
479  * a full-duplex bus. The function does not return until the transfer is complete.
480  *
481  *END*/
482 dspi_status_t DSPI_DRV_EdmaMasterTransferBlocking(uint32_t instance,
483  const dspi_edma_device_t * device,
484  const uint8_t * sendBuffer,
485  uint8_t * receiveBuffer,
486  size_t transferByteCount,
487  uint32_t timeout)
488 {
490  dspi_edma_master_state_t * dspiEdmaState = (dspi_edma_master_state_t *)g_dspiStatePtr[instance];
491  SPI_Type *base = g_dspiBase[instance];
492  dspi_status_t error = kStatus_DSPI_Success;
493 
494  dspiEdmaState->isTransferBlocking = true;
497  if (transferByteCount == 0)
498  {
499  return error;
500  }
501 
505  if (!FSL_FEATURE_DSPI_HAS_SEPARATE_DMA_RX_TX_REQn(instance))
506  {
507  if (dspiEdmaState->bitsPerFrame > 8)
508  {
509  if (transferByteCount > 1022)
510  {
511  return kStatus_DSPI_OutOfRange;
512  }
513  }
514  else
515  {
516  if (transferByteCount > 511)
517  {
518  return kStatus_DSPI_OutOfRange;
519  }
520  }
521  }
522 
524  osa_status_t syncStatus;
525 
526  if (DSPI_DRV_EdmaMasterStartTransfer(instance, device, sendBuffer, receiveBuffer,
527  transferByteCount) == kStatus_DSPI_Busy)
528  {
529  return kStatus_DSPI_Busy;
530  }
531 
533  do
534  {
535  syncStatus = OSA_SemaWait(&dspiEdmaState->irqSync, timeout);
536  }while(syncStatus == kStatus_OSA_Idle);
537 
541  if (syncStatus != kStatus_OSA_Success)
542  {
544  dspiEdmaState->isTransferInProgress = false;
545 
547  DSPI_HAL_SetRxFifoDrainDmaIntMode(base, kDspiGenerateDmaReq, false);
548 
550  DSPI_HAL_SetTxFifoFillDmaIntMode(base, kDspiGenerateDmaReq, false);
551 
553  DSPI_HAL_SetIntMode(base, kDspiEndOfQueue, false);
554 
555  error = kStatus_DSPI_Timeout;
556  }
557 
558  return error;
559 }

563  * Function Name : DSPI_DRV_EdmaMasterTransfer
564  * Description : Performs a non-blocking SPI master mode transfer with EDMA support.

585  {
586  return kStatus_DSPI_Success;

614  return kStatus_DSPI_Busy;
615  }
616 
618  return kStatus_DSPI_Success;

677 static dspi_status_t DSPI_DRV_EdmaMasterStartTransfer(uint32_t instance,
678  const dspi_edma_device_t * device,
679  const uint8_t * sendBuffer,
680  uint8_t * receiveBuffer,
681  size_t transferByteCount)
682 {
684  dspi_edma_master_state_t * dspiEdmaState = (dspi_edma_master_state_t *)g_dspiStatePtr[instance];
686  DMA_Type * dmaBaseAddr;
687  uint32_t dmaChannel;
688  uint32_t calculatedBaudRate;
689  dspi_command_config_t command;
690  SPI_Type *base = g_dspiBase[instance];
691  edma_transfer_config_t config;
692  uint32_t txTransferByteCnt = 0;
693  uint32_t rxTransferByteCnt = 0;
694 
696  s_wordToSend = 0;
697 
699  if (dspiEdmaState->isTransferInProgress)
700  {
701  return kStatus_DSPI_Busy;
702  }
703 
709  if (device)
710  {
711  DSPI_DRV_EdmaMasterConfigureBus(instance, device, &calculatedBaudRate);
712  dspiEdmaState->bitsPerFrame = device->dataBusConfig.bitsPerFrame; /*update bits/frame*/
713  }
714 
722  if ((dspiEdmaState->bitsPerFrame > 8) && (transferByteCount & 1UL))
723  {
724  dspiEdmaState->extraByte = true;
725  txTransferByteCnt = transferByteCount + 1U;
726  rxTransferByteCnt = transferByteCount & ~1U;
727  }
728  else
729  {
730  dspiEdmaState->extraByte = false;
731  txTransferByteCnt = transferByteCount;
732  rxTransferByteCnt = transferByteCount;
733  }
737  dspiEdmaState->rxBuffer = (uint8_t *)receiveBuffer;
738  dspiEdmaState->rxTransferByteCnt = rxTransferByteCnt;
739 
741  dspiEdmaState->isTransferInProgress = true;
742 
747  DSPI_HAL_PresetTransferCount(base, 0);
748 
750  DSPI_HAL_SetFlushFifoCmd(base, true, true);
751 
753  DSPI_HAL_ClearStatusFlag(base, kDspiTxComplete);
754  DSPI_HAL_ClearStatusFlag(base, kDspiEndOfQueue);
755  DSPI_HAL_ClearStatusFlag(base, kDspiTxFifoUnderflow);
756  DSPI_HAL_ClearStatusFlag(base, kDspiTxFifoFillRequest);
757  DSPI_HAL_ClearStatusFlag(base, kDspiRxFifoOverflow);
