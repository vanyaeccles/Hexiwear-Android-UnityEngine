384 static void HostInterface_RxCallback(
385  uint32_t instance,
386  void* uartState
387  )
388 {
389  static uint8_t
390  byteCount;
391 
392  uart_state_t*
393  state = (uart_state_t*)uartState;
394 
395  switch ( hostInterface_rxState )
396  {
397  case hostInterface_rxState_idle: {
398  if ( gHostInterface_startByte1 == hostInterface_rxPacket.start1 )
399  {
400  hostInterface_rxState++;
401  state->rxBuff++;
402  }
403 
404  break;
405  }
406 
407  case hostInterface_rxState_headerReceived: {
408  if (
409  ( gHostInterface_startByte1 != hostInterface_rxPacket.start1 )
410  || ( gHostInterface_startByte2 != ( hostInterface_rxPacket.start2 & 0xFE ) )
411  || ( hostInterface_rxPacket.length > gHostInterface_dataSize )
412  )
413  {
414  // return to idle state
415  hostInterface_rxState = hostInterface_rxState_idle;
416  // reset the RX buffer pointer
417  state->rxBuff = (uint8_t*) &hostInterface_rxPacket;
418  }
419 
420  else
421  {
422  // advance the buffer pointer
423  state->rxBuff++;
424 
425  if ( 0 == hostInterface_rxPacket.length )
426  {
427  // advance to "wait-trailer-bit" state
428  hostInterface_rxState = hostInterface_rxState_trailerWait;
429  }
430 
431  else
432  {
433  byteCount = 0;
434  // advance to "data-wait" state
435  hostInterface_rxState = hostInterface_rxState_dataWait;
436  }
437  }
438 
439  break;
440  }
441 
442  case hostInterface_rxState_dataWait: {
443  byteCount++;
444  (state->rxBuff)++;
445 
446  // if we reached the byte-count
447  if ( hostInterface_rxPacket.length == byteCount )
448  {
449  // advance to "wait-trailer-bit" state
450  hostInterface_rxState = hostInterface_rxState_trailerWait;
451  }
452 
453  break;
454  }
455 
456  case hostInterface_rxState_trailerWait: {
457  if ( gHostInterface_trailerByte == *(state->rxBuff) )
458  {
459 
460 #if defined (gHostInterface_RxConfirmationEnable)
461  if ( packetType_OK == hostInterface_rxPacket.type )
462  {
463  // announce that OK packet is received successfully
464  HostInterface_EventConfirmPacketSet();
465  }
466 
467  else
468 #endif
469  {
470  // else, put the packet into the RX queue
471  HostInterface_RxQueueMsgPut( (hostInterface_packet_t *)&hostInterface_rxPacket );
472  }
473  }
474 
475  // go back to the idle state
476  hostInterface_rxState = hostInterface_rxState_idle;
477  // reset the RX buffer pointer
478  state->rxBuff = (uint8_t *) &hostInterface_rxPacket;
479  break;
480  }
481 
482  default: {
483  hostInterface_rxState++;
484  state->rxBuff++;
485  }
486  }
487 }
